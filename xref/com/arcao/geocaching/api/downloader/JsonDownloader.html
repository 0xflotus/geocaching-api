1   package com.arcao.geocaching.api.downloader;
2   
3   import com.arcao.geocaching.api.exception.InvalidResponseException;
4   import com.arcao.geocaching.api.exception.NetworkException;
5   import com.arcao.geocaching.api.parser.JsonReader;
6   
7   import java.net.URL;
8   
9   /**
10   * Interface to implement downloading JSON data with GET and POST method
11   *
12   * @author arcao
13   */
14  public interface JsonDownloader {
15      /**
16       * Send request to a specified url with GET method and returns instance of
17       * {@link JsonReader} with a response
18       *
19       * @param url url to request including query parameters
20       * @return instance of {@link JsonReader} object with a response
21       * @throws NetworkException         If Network I/O error occurs
22       * @throws InvalidResponseException If response contains invalid data or HTTP response code isn't 200 (OK)
23       */
24      JsonReader get(URL url) throws NetworkException, InvalidResponseException;
25  
26      /**
27       * Send request to a specified url with POST method including post data and returns
28       * instance of {@link JsonReader} with a response
29       *
30       * @param url      url to request including query parameters
31       * @param postData post data
32       * @return instance of {@link JsonReader} object with a response
33       * @throws NetworkException         If Network I/O error occurs
34       * @throws InvalidResponseException If response contains invalid data or HTTP response code isn't 200 (OK)
35       */
36      JsonReader post(URL url, byte[] postData) throws NetworkException, InvalidResponseException;
37  }
