11  import org.jetbrains.annotations.NotNull;
12  import org.jetbrains.annotations.Nullable;
13  
14  import java.util.Collections;
15  import java.util.List;
16  
17  /**
18   * Abstract class of Geocaching API
19   *
20   * @author arcao
21   * @since 1.4.3
22   */
23  abstract class AbstractGeocachingApi implements GeocachingApi {
24      @Nullable String session;
25  
26      @Nullable
27      @Override
28      public String getSession() {
29          return session;
30      }
31  
32      @Override
33      public void openSession(@NotNull String session) throws GeocachingApiException {
34          this.session = session;
35      }
36  
37      @Override
38      public Geocache getGeocache(@NotNull ResultQuality resultQuality, @NotNull String cacheCode, int cacheLogCount, int trackableLogCount) throws GeocachingApiException {
39          final List<Geocache> list = searchForGeocaches(resultQuality, 1, cacheLogCount, trackableLogCount,
40                  Collections.singletonList((Filter) new CacheCodeFilter(cacheCode)), null);
41  
42          if (list.isEmpty())
43              return null;
44  
45          return list.get(0);
46      }
47  
48      @Override
49      public GeocacheLog createFieldNoteAndPublish(@NotNull FieldNote fieldNote, boolean publish, ImageData imageData, boolean favoriteThisCache) throws GeocachingApiException {
50          return createFieldNoteAndPublish(fieldNote.cacheCode(), fieldNote.logType(), fieldNote.dateLogged(), fieldNote.note(), publish, imageData,
51                  favoriteThisCache);
52      }
53  }
