3   import org.jetbrains.annotations.Nullable;

5   import java.util.Date;
6   import java.util.Locale;
7   
8   /**
9    * Helper class during JSON serializing process
10   *
11   * @author arcao
12   */
13  public final class JsonBuilder {
14      private JsonBuilder() {
15      }
16  
17      /**
18       * Convert Date object to JSON date notation
19       *
20       * @param date date convert
21       * @return JSON notation for date
22       */
23      public static String dateToJsonString(@Nullable Date date) {
24          if (date == null)
25              return null;
26  
27          return String.format(Locale.US, "/Date(%d)/", date.getTime());
28      }
29  }
