7   import java.io.Serializable;
8   import java.text.ParseException;
9   import java.text.SimpleDateFormat;
10  import java.util.Date;
11  import java.util.TimeZone;
12  
13  @AutoValue
14  public abstract class FieldNote implements Serializable {
15  	private static final long serialVersionUID = 249144828657285091L;
16  
17  	private static final SimpleDateFormat DATE_FORMAT = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
18  	static {
19  		DATE_FORMAT.setTimeZone(TimeZone.getTimeZone("UTC"));
20  	}
21  
22  	public static FieldNote parseLine(String line) {
23  		String[] items = line.split(",", 4);
24  
25  		String note = items[3];
26  		if (note.length() >= 2 && note.charAt(0) == '\"' && note.charAt(note.length() - 1) == '\"')
27  			note = note.substring(1, note.length() - 1);
28  
29  		try {
30  			return builder()
31  					.cacheCode(items[0])
32  					.dateLogged(DATE_FORMAT.parse(items[1]))
33  					.logType(GeocacheLogType.fromName(items[2]))
34  					.note(note)
35  					.build();
36  		} catch (ParseException e) {
37  			return null;
38  		}
39  	}
40  
41  	private static String safeNote(String note) {
42  		return note.replace('"', '\'').replaceAll("[\r\n\t]+", "");
43  	}
44  
45  	public static Builder builder() {
46  		return new AutoValue_FieldNote.Builder();
47  	}
48  
49  	public abstract String cacheCode();
50  	public abstract Date dateLogged();
51  	public abstract GeocacheLogType logType();
52  	public abstract String note();
53  
54  	@Override
55  	public String toString() {
56  		return String.format("%s,%s,%s,\"%s\"", cacheCode(), DATE_FORMAT.format(dateLogged()), logType().name, safeNote(note()));
57  	}
58  
59  	@AutoValue.Builder
60  	public abstract static class Builder {
61  		public abstract Builder cacheCode(String cacheCode);
62  		public abstract Builder dateLogged(Date dateLogged);
63  		public abstract Builder logType(GeocacheLogType logType);
64  		public abstract Builder note(String note);
65  		public abstract FieldNote build();
66  	}
67  }
