12  @AutoValue

14      private static final long serialVersionUID = 5480568653613770776L;
15  
16      /**
17       * Returns a cache limit
18       *
19       * @return limit
20       */
21      public abstract long limit();
22  
23      /**
24       * Returns a period for this limit in the minutes
25       *
26       * @return period in minutes
27       */
28      public abstract long period();
29  
30      public static Builder builder() {

32      }
33
