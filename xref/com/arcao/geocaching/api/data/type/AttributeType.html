148     public final String name;

156     @NotNull
157     public final String imageUrl;
158 
159     AttributeType(int id, @NotNull String name, boolean on, @NotNull String imageUrl) {
160         this.id = id;
161         this.name = name;
162         this.on = on;
163         this.imageUrl = imageUrl;
164     }
165 
166     @Override
167     public String toString() {
168         return name + (on ? " (ON)" : " (OFF)");
169     }
170 
171     /**
172      * Get an AttributeType from Groundspeak Id. If Groundspeak Id is not valid, the null is returned.
173      *
174      * @param id Groundspeak Id
175      * @param on is attribute On?
176      * @return Attribute type or null
177      */
178     @Nullable
179     public static AttributeType fromId(int id, boolean on) {
180         for (AttributeType type : values()) {
181             if (type.id == id && type.on == on)
182                 return type;
183         }
184 
185         return null;
186     }
