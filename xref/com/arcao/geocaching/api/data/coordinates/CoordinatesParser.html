19  public final class CoordinatesParser {
20      private static final Logger logger = LoggerFactory.getLogger(CoordinatesParser.class);
21      private static final double MINUTES_PER_DEGREE = 60.0;
22      private static final double SECONDS_PER_DEGREE = 3600.0;
23  
24      //                                                                   ( 1 )     ( 2 )          ( 3 )        ( 4 )        ( 5 )
25      private static final Pattern LATITUDE_PATTERN = Pattern.compile("\\b([NS])\\s*(\\d+)째?(?:\\s*(\\d+)(?:[.,](\\d+)|'?\\s*(\\d+(?:[.,]\\d+)?)(?:''|\")?)?)?", Pattern.CASE_INSENSITIVE);
26      private static final Pattern LONGITUDE_PATTERN = Pattern.compile("\\b([WE])\\s*(\\d+)째?(?:\\s*(\\d+)(?:[.,](\\d+)|'?\\s*(\\d+(?:[.,]\\d+)?)(?:''|\")?)?)?", Pattern.CASE_INSENSITIVE);
27  
28      //                                                                                                                   ( 1 )      ( 2 )            ( 3 )        ( 4 )        ( 5 )
29      private static final Pattern LATITUDE_PATTERN_UNSAFE = Pattern.compile("(?:(?=[\\-\\w])(?<![\\-\\w])|(?<![^\\-\\w]))([NS]|)\\s*(-?\\d+)째?(?:\\s*(\\d+)(?:[.,](\\d+)|'?\\s*(\\d+(?:[.,]\\d+)?)(?:''|\")?)?)?", Pattern.CASE_INSENSITIVE);
30      private static final Pattern LONGITUDE_PATTERN_UNSAFE = Pattern.compile("(?:(?=[\\-\\w])(?<![\\-\\w])|(?<![^\\-\\w]))([WE]|)\\s*(-?\\d+)째?(?:\\s*(\\d+)(?:[.,](\\d+)|'?\\s*(\\d+(?:[.,]\\d+)?)(?:''|\")?)?)?", Pattern.CASE_INSENSITIVE);
31  
32      private CoordinatesParser() {
33      }
34  
35      private enum CoordinateType {
36          LAT,
37          LON,
38          LAT_UNSAFE,
39          LON_UNSAFE
40      }
41  
42      public static Coordinates parse(String latitude, String longitude) throws ParseException {
43          return parse(latitude, longitude, true);
44      }
45  
46      public static Coordinates parse(String latitude, String longitude, boolean safe) throws ParseException {
47          return Coordinates.create(
48                  parseLatitude(latitude, safe),
49                  parseLongitude(longitude, safe)
50          );
51      }
52  
53      public static double parseLatitude(String latitude) throws ParseException {
54          return parseLatitude(latitude, true);
55      }
56  
57      public static double parseLatitude(String latitude, boolean safe) throws ParseException {
58          return parse(latitude, safe ? CoordinateType.LAT : CoordinateType.LAT_UNSAFE).result;
59      }
60  
61      public static double parseLongitude(String longitude) throws ParseException {
62          return parseLongitude(longitude, true);
63      }
64  
65      public static double parseLongitude(String longitude, boolean safe) throws ParseException {
66          return parse(longitude, safe ? CoordinateType.LON : CoordinateType.LON_UNSAFE).result;
67      }
68  
69      public static Coordinates parse(String coordinates) throws ParseException {
70          final ParseResult latitudeWrapper = parse(coordinates, CoordinateType.LAT);
71          final double lat = latitudeWrapper.result;
72          // cut away the latitude part when parsing the longitude
73          final ParseResult longitudeWrapper = parse(coordinates.substring(latitudeWrapper.matcherPos + latitudeWrapper.matcherLen), CoordinateType.LON);
74  
75          if (longitudeWrapper.matcherPos - (latitudeWrapper.matcherPos + latitudeWrapper.matcherLen) >= 10) {
76              throw new ParseException("Distance between latitude and longitude text is to large.", latitudeWrapper.matcherPos + latitudeWrapper.matcherLen + longitudeWrapper.matcherPos);
77          }
78  
79          final double lon = longitudeWrapper.result;
80          return Coordinates.create(lat, lon);
81      }
82  
83      protected static ParseResult parse(String coordinate, CoordinateType coordinateType) throws ParseException {
84          Pattern pattern;
85  
86          switch (coordinateType) {
87              case LAT_UNSAFE:
88                  pattern = LATITUDE_PATTERN_UNSAFE;
89                  break;
90              case LON_UNSAFE:
91                  pattern = LONGITUDE_PATTERN_UNSAFE;
92                  break;
93              case LON:
94                  pattern = LONGITUDE_PATTERN;
95                  break;
96              case LAT:
97              default:
98                  pattern = LATITUDE_PATTERN;
99                  break;
100         }
101 
102         final Matcher matcher = pattern.matcher(coordinate);
103 
104         if (matcher.find()) {
105             double sign = "S".equalsIgnoreCase(matcher.group(1)) || "W".equalsIgnoreCase(matcher.group(1)) ? -1.0 : 1.0;
106             double degree = Double.parseDouble(matcher.group(2));
107 
108             if (degree < 0) {
109                 sign = -1;
110                 degree = Math.abs(degree);
111             }
112 
113             double minutes = 0.0;
114             double seconds = 0.0;
115 
116             if (matcher.group(3) != null) {
117                 minutes = Double.parseDouble(matcher.group(3));
118 
119                 if (matcher.group(4) != null) {
120                     seconds = Double.parseDouble("0." + matcher.group(4)) * MINUTES_PER_DEGREE;
121                 } else if (matcher.group(5) != null) {
122                     seconds = Double.parseDouble(matcher.group(5).replace(",", "."));
123                 }
124             }
125 
126             double result = sign * (degree + minutes / MINUTES_PER_DEGREE + seconds / SECONDS_PER_DEGREE);
127 
128             // normalize result
129             switch (coordinateType) {
130                 case LON_UNSAFE:
131                 case LON:
132                     result = normalize(result, -180, 180);
133                     break;
134                 case LAT_UNSAFE:
135                 case LAT:
136                 default:
137                     result = normalize(result, -90, 90);
138                     break;
139             }
140 
141             return new ParseResult(result, matcher.start(), matcher.group().length());
142         } else {
143 
144             // Nothing found with "N 52...", try to match string as decimaldegree
145             try {
146                 final String[] items = StringUtils.split(coordinate.trim());
147                 if (items.length > 0) {
148                     final int index = (coordinateType == CoordinateType.LON ? items.length - 1 : 0);
149                     final int pos = (coordinateType == CoordinateType.LON ? coordinate.lastIndexOf(items[index]) : coordinate.indexOf(items[index]));
150                     return new ParseResult(Double.parseDouble(items[index]), pos, items[index].length());
151                 }
152             } catch (NumberFormatException e) {
153                 logger.error(e.getMessage(), e);
154             }
155         }
156         throw new ParseException("Could not parse coordinate: \"" + coordinate + "\"", 0);
157     }
158 
159     /**
160      * Normalizes any number to an arbitrary range by assuming the range wraps around when going below min or above max
161      *
162      * @param value input
163      * @param start range start
164      * @param end   range end
165      * @return normalized number
166      */
167     private static double normalize(double value, double start, double end) {
168         final double width = end - start;
169         final double offsetValue = value - start; // value relative to 0
170 
171         return (offsetValue - (Math.floor(offsetValue / width) * width)) + start; // + start to reset back to start of original range
172     }
173 
174     private static class ParseResult {
175         final double result;
176         final int matcherPos;
177         final int matcherLen;
178 
179         ParseResult(double result, int matcherPos, int matcherLen) {
180             this.result = result;
181             this.matcherPos = matcherPos;
182             this.matcherLen = matcherLen;
183         }
184     }
