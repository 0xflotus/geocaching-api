3   import com.google.gson.stream.JsonWriter;
4   
5   import org.jetbrains.annotations.NotNull;
6   
7   import java.io.IOException;
8   
9   public class NotFoundByUsersFilter implements Filter {
10      private static final String NAME = "NotFoundByUsers";
11  
12      @NotNull private final String[] userNames;
13  
14      public NotFoundByUsersFilter(@NotNull String... userNames) {
15          this.userNames = userNames;
16      }
17  
18      @NotNull
19      @Override
20      public String name() {
21          return NAME;
22      }
23  
24      @Override
25      public boolean valid() {
26          if (userNames.length == 0)
27              return false;
28  
29          for (String userName : userNames) {
30              if (userName != null && !userName.isEmpty())
31                  return true;
32          }
33  
34          return false;
35      }
36  
37      @Override
38      public void writeJson(@NotNull JsonWriter w) throws IOException {
39          w.name(NAME);
40          w.beginObject();
41          w.name("UserNames");
42          w.beginArray();
43          for (String userName : userNames) {
44              if (userName != null && !userName.isEmpty())
45                  w.value(userName);
46          }
47          w.endArray();
48          w.endObject();
49      }
50  }
