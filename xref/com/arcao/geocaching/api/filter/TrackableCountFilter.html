1   package com.arcao.geocaching.api.filter;
2   
3   import com.google.gson.stream.JsonWriter;
4   
5   import org.jetbrains.annotations.NotNull;
6   
7   import java.io.IOException;
8   
9   public class TrackableCountFilter implements Filter {
10      private static final String NAME = "TrackableCount";
11  
12      private final Integer min;
13      private final Integer max;
14  
15      public TrackableCountFilter(Integer min, Integer max) {
16          this.min = min;
17          this.max = max;
18      }
19  
20      @NotNull
21      @Override
22      public String name() {
23          return NAME;
24      }
25  
26      @Override
27      public boolean valid() {
28          return !(min != null && max != null && min > max) && (min != null || max != null);
29  
30      }
31  
32      @Override
33      public void writeJson(@NotNull JsonWriter w) throws IOException {
34          w.name(NAME);
35          w.beginObject();
36          if (min != null)
37              w.name("MinTrackables").value(min.intValue());
38          if (max != null)
39              w.name("MaxTrackables").value(max.intValue());
40          w.endObject();
41      }
42  }
