4   import com.arcao.geocaching.api.data.type.GeocacheType;
5   import com.google.gson.stream.JsonWriter;
6   
7   import org.jetbrains.annotations.NotNull;
8   
9   import java.io.IOException;
10  
11  public class GeocacheTypeFilter implements Filter {
12      private static final String NAME = "GeocacheType";
13  
14      @NotNull private final GeocacheType[] geocacheTypes;
15  
16      public GeocacheTypeFilter(@NotNull GeocacheType... geocacheTypes) {
17          this.geocacheTypes = geocacheTypes;
18      }
19  
20      @NotNull
21      @Override
22      public String name() {
23          return NAME;
24      }
25  
26      @Override
27      public boolean valid() {
28          if (geocacheTypes.length == 0)
29              return false;
30  
31          boolean valid = false;
32          for (GeocacheType geocacheType : geocacheTypes) {
33              if (geocacheType != null)
34                  valid = true;
35          }
36  
37          return valid;
38      }
39  
40      @Override
41      public void writeJson(@NotNull JsonWriter w) throws IOException {
42          w.name(NAME);
43          w.beginObject();
44          w.name("GeocacheTypeIds");
45          w.beginArray();
46          for (GeocacheType geocacheType : geocacheTypes) {
47              if (geocacheType != null)
48                  w.value(geocacheType.id);
49          }
50          w.endArray();
51          w.endObject();
52      }
53  }
