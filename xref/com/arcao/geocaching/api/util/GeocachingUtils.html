3   /**
4    * Set of Geocaching helper methods
5    *
6    * @author arcao
7    * @since 1.5
8    */
9   public final class GeocachingUtils {
10      private static final String BASE_31_CHARS = "0123456789ABCDEFGHJKMNPQRTVWXYZ";
11  
12      // = (16 * 31 * 31 * 31) - (16 * 16 * 16 * 16)
13      private static final long CACHE_CODE_BASE31_MAGIC_NUMBER = 411120;
14      private static final String CACHE_PREFIX = "GC";
15      private static final long CACHE_CODE_BASE16_MAX = 0xFFFF;
16      private static final int BASE_16 = 16;
17  
18      private GeocachingUtils() {
19      }
20  
21      /**
22       * Convert a base 31 number containing chars 0123456789ABCDEFGHJKMNPQRTVWXYZ
23       * to numeric value.
24       *
25       * @param input base 31 number
26       * @return numeric value
27       * @throws IllegalArgumentException If input contains illegal chars
28       */
29      public static long base31Decode(final String input) {
30          final int base = BASE_31_CHARS.length();
31          long ret = 0;
32  
33          for (char ch : input.toCharArray()) {
34              ret *= base;
35  
36              int index = BASE_31_CHARS.indexOf(ch);
37              if (index == -1)
38                  throw new IllegalArgumentException("Only chars " + BASE_31_CHARS + " are supported.");
39  
40              ret += index;
41          }
42          return ret;
43      }
44  
45      /**
46       * Convert a numeric value to base 31 number using chars
47       * 0123456789ABCDEFGHJKMNPQRTVWXYZ.
48       *
49       * @param input numeric value
50       * @return base 31 number
51       */
52      public static String base31Encode(long input) {
53          final int base = BASE_31_CHARS.length();
54          StringBuilder sb = new StringBuilder();
55  
56          while (input != 0) {
57              sb.append(BASE_31_CHARS.charAt((int) (input % base)));
58              input /= base;
59          }
60  
61          return sb.reverse().toString();
62      }
63  
64      /**
65       * Convert cache code GCxxx to numeric cache id.<br>
66       * <br>
67       * The algorithm respects following rules used for cache code:<br>
68       * <ul>
69       * <li>GC0 - GCFFFF = value after GC prefix is a hexadecimal number
70       * <li>GCG000 - ... = value after GC is a base 31 number minus magic constant
71       * 411120 (16 * 31 * 31 * 31 - 16 * 16 * 16 * 16)
72       * </ul>
73       *
74       * @param cacheCode cache code including GC prefix
75       * @return cache id
76       * @throws IllegalArgumentException Cache code does not contains GC prefix or contains invalid characters
77       * @see #base31Decode(String)
78       */
79      public static long cacheCodeToCacheId(final String cacheCode) {
80          String cacheCodeNorm = cacheCode.toUpperCase();
81  
82          if (cacheCodeNorm.length() < 3 || !cacheCodeNorm.startsWith(CACHE_PREFIX))
83              throw new IllegalArgumentException("Cache code is too short or does not contains GC prefix.");
84  
85          // remove GC prefix
86          String code = cacheCodeNorm.substring(2);
87  
88          // 0 - FFFF = base16; G000 - ... = base 31
89          if (code.length() <= 4 && code.charAt(0) < 'G') {
90              try {
91                  return Long.parseLong(code, BASE_16);
92              } catch (NumberFormatException e) {
93                  throw new IllegalArgumentException("Only chars " + BASE_31_CHARS + " are supported.");
94              }
95          } else {
96              return base31Decode(code) - CACHE_CODE_BASE31_MAGIC_NUMBER;
97          }
98      }
99  
100     /**
101      * Convert a numeric cache id to cache code GCxxx. The algorithm respects
102      * rules for generating cache code.
103      *
104      * @param cacheId cache id
105      * @return cache code including GC prefix
106      * @see #cacheCodeToCacheId(String)
107      * @see #base31Encode(long)
108      */
109     public static String cacheIdToCacheCode(long cacheId) {
110         StringBuilder sb = new StringBuilder();
111 
112         // append GC prefix
113         sb.append(CACHE_PREFIX);
114 
115         if (cacheId <= CACHE_CODE_BASE16_MAX) { // 0 - FFFF
116             sb.append(Long.toString(cacheId, BASE_16).toUpperCase());
117         } else { // G000 - ...
118             sb.append(base31Encode(cacheId + CACHE_CODE_BASE31_MAGIC_NUMBER));
119         }
120 
121         return sb.toString();
122     }
123 
124     /**
125      * Returns true if the cache code is valid. The algorithm respects
126      * rules for cache code.
127      *
128      * @param cacheCode cache code
129      * @return true if cache code is valid, otherwise false
130      * @see #cacheCodeToCacheId(String)
131      */
132     public static boolean isCacheCodeValid(String cacheCode) {
133         try {
134             return cacheCodeToCacheId(cacheCode) >= 0;
135         } catch (IllegalArgumentException e) {
136             return false;
137         }
138     }
