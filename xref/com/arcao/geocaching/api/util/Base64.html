20  /**
21   * Utilities for encoding and decoding the Base64 representation of
22   * binary data.  See RFCs <a
23   * href="http://www.ietf.org/rfc/rfc2045.txt">2045</a> and <a
24   * href="http://www.ietf.org/rfc/rfc3548.txt">3548</a>.
25   */
26  public final class Base64 {
27      /**
28       * Default values for encoder/decoder flags.
29       */
30      public static final int DEFAULT = 0;
31   
32      /**
33       * Encoder flag bit to omit the padding '=' characters at the end
34       * of the output (if any).
35       */
36      public static final int NO_PADDING = 1;
37   
38      /**
39       * Encoder flag bit to omit all line terminators (i.e., the output
40       * will be on one long line).
41       */
42      public static final int NO_WRAP = 2;
43   
44      /**
45       * Encoder flag bit to indicate lines should be terminated with a
46       * CRLF pair instead of just an LF.  Has no effect if {@code
47       * NO_WRAP} is specified as well.
48       */
49      public static final int CRLF = 4;
50   
51      /**
52       * Encoder/decoder flag bit to indicate using the "URL and
53       * filename safe" variant of Base64 (see RFC 3548 section 4) where
54       * {@code -} and {@code _} are used in place of {@code +} and
55       * {@code /}.
56       */
57      public static final int URL_SAFE = 8;
58   
59      /**
60       * Flag to pass to {@link Base64OutputStream} to indicate that it
61       * should not close the output stream it is wrapping when it
62       * itself is closed.
63       */
64      public static final int NO_CLOSE = 16;

70      /* package */ static abstract class Coder {
71          public byte[] output;
72          public int op;
73   
74          /**
75           * Encode/decode another block of input data.  this.output is
76           * provided by the caller, and must be big enough to hold all
77           * the coded data.  On exit, this.opwill be set to the length
78           * of the coded data.
79           * @param input 
80           * @param offset 
81           * @param len 
82           *
83           * @param finish true if this is the final call to process for
84           *        this object.  Will finalize the coder state and
85           *        include any final bytes in the output.
86           *
87           * @return true if the input so far is good; false if some
88           *         error has been detected in the input stream..
89           */
90          public abstract boolean process(byte[] input, int offset, int len, boolean finish);
91   
92          /**
93           * @param len 
94           * @return the maximum number of bytes a call to process()
95           * could produce for the given number of input bytes.  This may
96           * be an overestimate.
97           */
98          public abstract int maxOutputSize(int len);
99      }

105     /**
106      * Decode the Base64-encoded data in input and return the data in
107      * a new byte array.
108      *
109      * <p>The padding '=' characters at the end are considered optional, but
110      * if any are present, there must be the correct number of them.
111      *
112      * @param str    the input String to decode, which is converted to
113      *               bytes using the default charset
114      * @param flags  controls certain features of the decoded output.
115      *               Pass {@code DEFAULT} to decode standard Base64.
116      * @return decoded data
117      *
118      * @throws IllegalArgumentException if the input contains
119      * incorrect padding
120      */
121     public static byte[] decode(String str, int flags) {
122         return decode(str.getBytes(), flags);
123     }

125     /**
126      * Decode the Base64-encoded data in input and return the data in
127      * a new byte array.
128      *
129      * <p>The padding '=' characters at the end are considered optional, but
130      * if any are present, there must be the correct number of them.
131      *
132      * @param input the input array to decode
133      * @param flags  controls certain features of the decoded output.
134      *               Pass {@code DEFAULT} to decode standard Base64.
135      * @return decoded data
136      *
137      * @throws IllegalArgumentException if the input contains
138      * incorrect padding
139      */
140     public static byte[] decode(byte[] input, int flags) {
141         return decode(input, 0, input.length, flags);
142     }

144     /**
145      * Decode the Base64-encoded data in input and return the data in
146      * a new byte array.
147      *
148      * <p>The padding '=' characters at the end are considered optional, but
149      * if any are present, there must be the correct number of them.
150      *
151      * @param input  the data to decode
152      * @param offset the position within the input array at which to start
153      * @param len    the number of bytes of input to decode
154      * @param flags  controls certain features of the decoded output.
155      *               Pass {@code DEFAULT} to decode standard Base64.
156      * @return decoded data
157      *
158      * @throws IllegalArgumentException if the input contains
159      * incorrect padding
160      */
161     public static byte[] decode(byte[] input, int offset, int len, int flags) {
162         // Allocate space for the most data the input could represent.
163         // (It could contain less if it contains whitespace, etc.)
164         Decoder decoder = new Decoder(flags, new byte[len*3/4]);
165  
166         if (!decoder.process(input, offset, len, true)) {
167             throw new IllegalArgumentException("bad base-64");
168         }
169  
170         // Maybe we got lucky and allocated exactly enough output space.
171         if (decoder.op == decoder.output.length) {
172             return decoder.output;
173         }
174  
175         // Need to shorten the array, so allocate a new one of the
176         // right size and copy.
177         byte[] temp = new byte[decoder.op];
178         System.arraycopy(decoder.output, 0, temp, 0, decoder.op);
179         return temp;
180     }

229         /** Non-data values in the DECODE arrays. */
230         private static final int SKIP = -1;
231         private static final int EQUALS = -2;
232  
233         /**
234          * States 0-3 are reading through the next input tuple.
235          * State 4 is having read one '=' and expecting exactly
236          * one more.
237          * State 5 is expecting no more data or padding characters
238          * in the input.
239          * State 6 is the error state; an error has been detected
240          * in the input and no future input can "fix" it.
241          */
242         private int state;   // state number (0 to 6)
243         private int value;
244  
245         final private int[] alphabet;
246  
247         public Decoder(int flags, byte[] output) {
248             this.output = output;
249  
250             alphabet = ((flags & URL_SAFE) == 0) ? DECODE : DECODE_WEBSAFE;
251             state = 0;
252             value = 0;
253         }

255         /**
256          * @return an overestimate for the number of bytes {@code
257          * len} bytes could decode to.
258          */
259         @Override
260 		public int maxOutputSize(int len) {
261             return len * 3/4 + 10;
262         }
263  
264         /**
265          * Decode another block of input data.
266          *
267          * @return true if the state machine is still healthy.  false if
268          *         bad base-64 data has been detected in the input stream.
269          */
270         @Override
271 		public boolean process(byte[] input, int offset, int len, boolean finish) {
272             if (this.state == 6) return false;
273  
274             int p = offset;
275             len += offset;
276  
277             // Using local variables makes the decoder about 12%
278             // faster than if we manipulate the member variables in
279             // the loop.  (Even alphabet makes a measurable
280             // difference, which is somewhat surprising to me since
281             // the member variable is final.)
282             int state = this.state;
283             int value = this.value;
284             int op = 0;
285             final byte[] output = this.output;
286             final int[] alphabet = this.alphabet;
287  
288             while (p < len) {
289                 // Try the fast path:  we're starting a new tuple and the
290                 // next four bytes of the input stream are all data
291                 // bytes.  This corresponds to going through states
292                 // 0-1-2-3-0.  We expect to use this method for most of
293                 // the data.
294                 //
295                 // If any of the next four bytes of input are non-data
296                 // (whitespace, etc.), value will end up negative.  (All
297                 // the non-data values in decode are small negative
298                 // numbers, so shifting any of them up and or'ing them
299                 // together will result in a value with its top bit set.)
300                 //
301                 // You can remove this whole block and the output should
302                 // be the same, just slower.
303                 if (state == 0) {
304                     while (p+4 <= len &&
305                            (value = ((alphabet[input[p] & 0xff] << 18) |
306                                      (alphabet[input[p+1] & 0xff] << 12) |
307                                      (alphabet[input[p+2] & 0xff] << 6) |
308                                      (alphabet[input[p+3] & 0xff]))) >= 0) {
309                         output[op+2] = (byte) value;
310                         output[op+1] = (byte) (value >> 8);
311                         output[op] = (byte) (value >> 16);
312                         op += 3;
313                         p += 4;
314                     }
315                     if (p >= len) break;
316                 }
317  
318                 // The fast path isn't available -- either we've read a
319                 // partial tuple, or the next four input bytes aren't all
320                 // data, or whatever.  Fall back to the slower state
321                 // machine implementation.
322  
323                 int d = alphabet[input[p++] & 0xff];
324  
325                 switch (state) {
326                 case 0:
327                     if (d >= 0) {
328                         value = d;
329                         ++state;
330                     } else if (d != SKIP) {
331                         this.state = 6;
332                         return false;
333                     }
334                     break;
335  
336                 case 1:
337                     if (d >= 0) {
338                         value = (value << 6) | d;
339                         ++state;
340                     } else if (d != SKIP) {
341                         this.state = 6;
342                         return false;
343                     }
344                     break;
345  
346                 case 2:
347                     if (d >= 0) {
348                         value = (value << 6) | d;
349                         ++state;
350                     } else if (d == EQUALS) {
351                         // Emit the last (partial) output tuple;
352                         // expect exactly one more padding character.
353                         output[op++] = (byte) (value >> 4);
354                         state = 4;
355                     } else if (d != SKIP) {
356                         this.state = 6;
357                         return false;
358                     }
359                     break;
360  
361                 case 3:
362                     if (d >= 0) {
363                         // Emit the output triple and return to state 0.
364                         value = (value << 6) | d;
365                         output[op+2] = (byte) value;
366                         output[op+1] = (byte) (value >> 8);
367                         output[op] = (byte) (value >> 16);
368                         op += 3;
369                         state = 0;
370                     } else if (d == EQUALS) {
371                         // Emit the last (partial) output tuple;
372                         // expect no further data or padding characters.
373                         output[op+1] = (byte) (value >> 2);
374                         output[op] = (byte) (value >> 10);
375                         op += 2;
376                         state = 5;
377                     } else if (d != SKIP) {
378                         this.state = 6;
379                         return false;
380                     }
381                     break;
382  
383                 case 4:
384                     if (d == EQUALS) {
385                         ++state;
386                     } else if (d != SKIP) {
387                         this.state = 6;
388                         return false;
389                     }
390                     break;
391  
392                 case 5:
393                     if (d != SKIP) {
394                         this.state = 6;
395                         return false;
396                     }
397                     break;
398                 }
399             }
400  
401             if (!finish) {
402                 // We're out of input, but a future call could provide
403                 // more.
404                 this.state = state;
405                 this.value = value;
406                 this.op = op;
407                 return true;
408             }
409  
410             // Done reading input.  Now figure out where we are left in
411             // the state machine and finish up.
412  
413             switch (state) {
414             case 0:
415                 // Output length is a multiple of three.  Fine.
416                 break;
417             case 1:
418                 // Read one extra input byte, which isn't enough to
419                 // make another output byte.  Illegal.
420                 this.state = 6;
421                 return false;
422             case 2:
423                 // Read two extra input bytes, enough to emit 1 more
424                 // output byte.  Fine.
425                 output[op++] = (byte) (value >> 4);
426                 break;
427             case 3:
428                 // Read three extra input bytes, enough to emit 2 more
429                 // output bytes.  Fine.
430                 output[op++] = (byte) (value >> 10);
431                 output[op++] = (byte) (value >> 2);
432                 break;
433             case 4:
434                 // Read one padding '=' when we expected 2.  Illegal.
435                 this.state = 6;
436                 return false;
437             case 5:
438                 // Read all the padding '='s we expected and no more.
439                 // Fine.
440                 break;
441             }
442  
443             this.state = state;
444             this.op = op;
445             return true;
446         }
447     }

453     /**
454      * Base64-encode the given data and return a newly allocated
455      * String with the result.
456      *
457      * @param input  the data to encode
458      * @param flags  controls certain features of the encoded output.
459      *               Passing {@code DEFAULT} results in output that
460      *               adheres to RFC 2045.
461      * @return encoded data
462      */
463     public static String encodeToString(byte[] input, int flags) {
464         try {
465             return new String(encode(input, flags), "US-ASCII");
466         } catch (UnsupportedEncodingException e) {
467             // US-ASCII is guaranteed to be available.
468             throw new AssertionError(e);
469         }
470     }

475      *

477      * @param offset the position within the input array at which to
478      *               start
479      * @param len    the number of bytes of input to encode

481      *               Passing {@code DEFAULT} results in output that
482      *               adheres to RFC 2045.
483      * @return encoded data

485     public static String encodeToString(byte[] input, int offset, int len, int flags) {
486         try {

494     /**
495      * Base64-encode the given data and return a newly allocated
496      * byte[] with the result.
497      *
498      * @param input  the data to encode
499      * @param flags  controls certain features of the encoded output.
500      *               Passing {@code DEFAULT} results in output that
501      *               adheres to RFC 2045.
502      * @return encoded data
503      */
504     public static byte[] encode(byte[] input, int flags) {
505         return encode(input, 0, input.length, flags);
506     }
507  
508     /**
509      * Base64-encode the given data and return a newly allocated
510      * byte[] with the result.
511      *
512      * @param input  the data to encode
513      * @param offset the position within the input array at which to
514      *               start
515      * @param len    the number of bytes of input to encode
516      * @param flags  controls certain features of the encoded output.
517      *               Passing {@code DEFAULT} results in output that
518      *               adheres to RFC 2045.
519      * @return encoded data
520      */
521     public static byte[] encode(byte[] input, int offset, int len, int flags) {
522         Encoder encoder = new Encoder(flags, null);
523  
524         // Compute the exact length of the array we will produce.
525         int output_len = len / 3 * 4;
526  
527         // Account for the tail of the data and the padding bytes, if any.
528         if (encoder.do_padding) {
529             if (len % 3 > 0) {
530                 output_len += 4;
531             }
532         } else {
533             switch (len % 3) {
534                 case 0: break;
535                 case 1: output_len += 2; break;
536                 case 2: output_len += 3; break;
537             }
538         }
539  
540         // Account for the newlines, if any.
541         if (encoder.do_newline && len > 0) {
542             output_len += (((len-1) / (3 * Encoder.LINE_GROUPS)) + 1) *
543                 (encoder.do_cr ? 2 : 1);
544         }
545  
546         encoder.output = new byte[output_len];
547         encoder.process(input, offset, len, true);
548  
549         assert encoder.op == output_len;
550  
551         return encoder.output;
552     }

605         }
606  
607         /**
608          * @return an overestimate for the number of bytes {@code
609          * len} bytes could encode to.
610          */
611         @Override
612 		public int maxOutputSize(int len) {
613             return len * 8/5 + 10;
614         }
615  
616         @Override
617 		public boolean process(byte[] input, int offset, int len, boolean finish) {
618             // Using local variables makes the encoder about 9% faster.
619             final byte[] alphabet = this.alphabet;
620             final byte[] output = this.output;
621             int op = 0;
622             int count = this.count;
623  
624             int p = offset;
625             len += offset;
626             int v = -1;
627  
628             // First we need to concatenate the tail of the previous call
629             // with any input bytes available now and see if we can empty
630             // the tail.
631  
632             switch (tailLen) {
633                 case 0:
634                     // There was no tail.
635                     break;
636  
637                 case 1:
638                     if (p+2 <= len) {
639                         // A 1-byte tail with at least 2 bytes of
640                         // input available now.
641                         v = ((tail[0] & 0xff) << 16) |
642                             ((input[p++] & 0xff) << 8) |
643                             (input[p++] & 0xff);
644                         tailLen = 0;
645                     }
646                     break;
647  
648                 case 2:
649                     if (p+1 <= len) {
650                         // A 2-byte tail with at least 1 byte of input.
651                         v = ((tail[0] & 0xff) << 16) |
652                             ((tail[1] & 0xff) << 8) |
653                             (input[p++] & 0xff);
654                         tailLen = 0;
655                     }
656                     break;
657             }
658  
659             if (v != -1) {
660                 output[op++] = alphabet[(v >> 18) & 0x3f];
661                 output[op++] = alphabet[(v >> 12) & 0x3f];
662                 output[op++] = alphabet[(v >> 6) & 0x3f];
663                 output[op++] = alphabet[v & 0x3f];
664                 if (--count == 0) {
665                     if (do_cr) output[op++] = '\r';
666                     output[op++] = '\n';
667                     count = LINE_GROUPS;
668                 }
669             }
670  
671             // At this point either there is no tail, or there are fewer
672             // than 3 bytes of input available.
673  
674             // The main loop, turning 3 input bytes into 4 output bytes on
675             // each iteration.
676             while (p+3 <= len) {
677                 v = ((input[p] & 0xff) << 16) |
678                     ((input[p+1] & 0xff) << 8) |
679                     (input[p+2] & 0xff);
680                 output[op] = alphabet[(v >> 18) & 0x3f];
681                 output[op+1] = alphabet[(v >> 12) & 0x3f];
682                 output[op+2] = alphabet[(v >> 6) & 0x3f];
683                 output[op+3] = alphabet[v & 0x3f];
684                 p += 3;
685                 op += 4;
686                 if (--count == 0) {
687                     if (do_cr) output[op++] = '\r';
688                     output[op++] = '\n';
689                     count = LINE_GROUPS;
690                 }
691             }
692  
693             if (finish) {
694                 // Finish up the tail of the input.  Note that we need to
695                 // consume any bytes in tail before any bytes
696                 // remaining in input; there should be at most two bytes
697                 // total.
698  
699                 if (p-tailLen == len-1) {
700                     int t = 0;
701                     v = ((tailLen > 0 ? tail[t++] : input[p++]) & 0xff) << 4;
702                     tailLen -= t;
703                     output[op++] = alphabet[(v >> 6) & 0x3f];
704                     output[op++] = alphabet[v & 0x3f];
705                     if (do_padding) {
706                         output[op++] = '=';
707                         output[op++] = '=';
708                     }
709                     if (do_newline) {
710                         if (do_cr) output[op++] = '\r';
711                         output[op++] = '\n';
712                     }
713                 } else if (p-tailLen == len-2) {
714                     int t = 0;
715                     v = (((tailLen > 1 ? tail[t++] : input[p++]) & 0xff) << 10) |
716                         (((tailLen > 0 ? tail[t++] : input[p++]) & 0xff) << 2);
717                     tailLen -= t;
718                     output[op++] = alphabet[(v >> 12) & 0x3f];
719                     output[op++] = alphabet[(v >> 6) & 0x3f];
720                     output[op++] = alphabet[v & 0x3f];
721                     if (do_padding) {
722                         output[op++] = '=';
723                     }
724                     if (do_newline) {
725                         if (do_cr) output[op++] = '\r';
726                         output[op++] = '\n';
727                     }
728                 } else if (do_newline && op > 0 && count != LINE_GROUPS) {
729                     if (do_cr) output[op++] = '\r';
730                     output[op++] = '\n';
731                 }
732  
733                 assert tailLen == 0;
734                 assert p == len;
735             } else {
736                 // Save the leftovers in tail to be consumed on the next
737                 // call to encodeInternal.
738  
739                 if (p == len-1) {
740                     tail[tailLen++] = input[p];
741                 } else if (p == len-2) {
742                     tail[tailLen++] = input[p];
743                     tail[tailLen++] = input[p+1];
744                 }
745             }
746  
747             this.op = op;
748             this.count = count;
749  
750             return true;
751         }
752     }
