12  import static com.arcao.geocaching.api.parser.JsonParserUtil.parseJsonUTCDate;

14  public final class UserWaypointsJsonParser {
15      private UserWaypointsJsonParser() {
16      }
17  
18      public static List<UserWaypoint> parseList(JsonReader r) throws IOException {
19          if (r.peek() != JsonToken.BEGIN_ARRAY) {
20              r.skipValue();
21          }
22  
23          List<UserWaypoint> list = new ArrayList<UserWaypoint>();
24          r.beginArray();
25          while (r.hasNext()) {
26              list.add(parse(r));
27          }
28          r.endArray();
29          return list;
30      }
31  
32      private static UserWaypoint parse(JsonReader r) throws IOException {
33          UserWaypoint.Builder builder = UserWaypoint.builder();
34          Coordinates.Builder coordinatesBuilder = Coordinates.builder();
35  
36          r.beginObject();
37          while (r.hasNext()) {
38              String name = r.nextName();
39              if ("CacheCode".equals(name)) {
40                  builder.cacheCode(r.nextString());
41              } else if ("Description".equals(name)) {
42                  builder.description(r.nextString());
43              } else if ("ID".equals(name)) {
44                  builder.id(r.nextLong());
45              } else if ("Latitude".equals(name)) {
46                  coordinatesBuilder.latitude(r.nextDouble());
47              } else if ("Longitude".equals(name)) {
48                  coordinatesBuilder.longitude(r.nextDouble());
49              } else if ("UTCDate".equals(name)) {
50                  builder.date(parseJsonUTCDate(r.nextString()));
51              } else if ("UserID".equals(name)) {
52                  builder.userId(r.nextInt());
53              } else if ("IsCorrectedCoordinate".equals(name)) {
54                  builder.correctedCoordinate(r.nextBoolean());
55              } else {
56                  r.skipValue();
57              }
58          }
59          r.endObject();
60  
61          builder.coordinates(coordinatesBuilder.build());
62  
63          return builder.build();
64      }
65  }
