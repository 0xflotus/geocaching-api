12  import static com.arcao.geocaching.api.data.type.WaypointType.fromName;
13  import static com.arcao.geocaching.api.parser.JsonParserUtil.parseJsonUTCDate;

15  public final class WaypointJsonParser {
16      private WaypointJsonParser() {
17      }
18  
19      public static List<Waypoint> parseList(JsonReader r) throws IOException {
20          if (r.peek() != JsonToken.BEGIN_ARRAY) {
21              r.skipValue();
22          }
23  
24          List<Waypoint> list = new ArrayList<Waypoint>();
25          r.beginArray();
26          while (r.hasNext()) {
27              list.add(parse(r));
28          }
29          r.endArray();
30          return list;
31      }
32  
33      private static Waypoint parse(JsonReader r) throws IOException {
34          Waypoint.Builder builder = Waypoint.builder();
35          Coordinates.Builder coordinatesBuilder = Coordinates.builder();
36  
37          r.beginObject();
38          while (r.hasNext()) {
39              String name = r.nextName();
40              if ("Longitude".equals(name)) {
41                  coordinatesBuilder.longitude(r.nextDouble());
42              } else if ("Latitude".equals(name)) {
43                  coordinatesBuilder.latitude(r.nextDouble());
44              } else if ("UTCEnteredDate".equals(name)) {
45                  builder.time(parseJsonUTCDate(r.nextString()));
46              } else if ("Code".equals(name)) {
47                  builder.waypointCode(r.nextString());
48              } else if ("Name".equals(name)) {
49                  builder.waypointType(fromName(r.nextString()));
50              } else if ("Description".equals(name)) {
51                  builder.name(r.nextString());
52              } else if ("Comment".equals(name)) {
53                  builder.note(r.nextString());
54              } else {
55                  r.skipValue();
56              }
57          }
58          r.endObject();
59  
60          builder.coordinates(coordinatesBuilder.build());
61  
62          return builder.build();
63      }
64  }
