3   import com.arcao.geocaching.api.data.apilimits.MaxPerPage;
4   
5   import java.io.IOException;
6   
7   public final class MaxPerPageJsonParser {
8       private MaxPerPageJsonParser() {
9       }
10  
11      public static MaxPerPage parse(JsonReader r) throws IOException {
12          MaxPerPage.Builder builder = MaxPerPage.builder();
13  
14          r.beginObject();
15          while (r.hasNext()) {
16              String name = r.nextName();
17              if ("Geocaches".equals(name)) {
18                  builder.geocaches(r.nextInt());
19              } else if ("GeocacheLogs".equals(name)) {
20                  builder.geocacheLogs(r.nextInt());
21              } else if ("Trackables".equals(name)) {
22                  builder.trackables(r.nextInt());
23              } else if ("TrackableLogs".equals(name)) {
24                  builder.trackableLogs(r.nextInt());
25              } else if ("CacheNotes".equals(name)) {
26                  builder.cacheNotes(r.nextInt());
27              } else if ("GalleryImages".equals(name)) {
28                  builder.galleryImages(r.nextInt());
29              } else {
30                  r.skipValue();
31              }
32          }
33          r.endObject();
34  
35          return builder.build();
36      }
37  }
