3   import com.arcao.geocaching.api.data.ImageData;
4   import com.google.gson.stream.JsonToken;
5   
6   import java.io.IOException;
7   import java.util.ArrayList;
8   import java.util.List;
9   
10  public final class ImageDataJsonParser {
11      private ImageDataJsonParser() {
12      }
13  
14      public static List<ImageData> parseList(JsonReader r) throws IOException {
15          if (r.peek() != JsonToken.BEGIN_ARRAY) {
16              r.skipValue();
17          }
18  
19          List<ImageData> list = new ArrayList<ImageData>();
20          r.beginArray();
21  
22          while (r.hasNext()) {
23              list.add(parse(r));
24          }
25          r.endArray();
26          return list;
27      }
28  
29      private static ImageData parse(JsonReader r) throws IOException {
30          ImageData.Builder builder = ImageData.builder();
31  
32          r.beginObject();
33          while (r.hasNext()) {
34              String name = r.nextName();
35              if ("Description".equals(name)) {
36                  builder.description(r.nextString());
37              } else if ("MobileUrl".equals(name)) {
38                  builder.mobileUrl(r.nextString());
39              } else if ("Name".equals(name)) {
40                  builder.name(r.nextString());
41              } else if ("ThumbUrl".equals(name)) {
42                  builder.thumbUrl(r.nextString());
43              } else if ("Url".equals(name)) {
44                  builder.url(r.nextString());
45              } else {
46                  r.skipValue();
47              }
48          }
49          r.endObject();
50  
51          return builder.build();
52      }
53  }
