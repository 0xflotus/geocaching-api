5   import java.io.IOException;
6   
7   public final class CacheLimitsJsonParser {
8       private CacheLimitsJsonParser() {
9       }
10  
11      public static GeocacheLimits parse(JsonReader r) throws IOException {
12          GeocacheLimits.Builder builder = GeocacheLimits.builder();
13  
14          r.beginObject();
15          while (r.hasNext()) {
16              String name = r.nextName();
17              if ("CachesLeft".equals(name)) {
18                  builder.geocacheLeft(r.nextInt());
19              } else if ("CurrentCacheCount".equals(name)) {
20                  builder.currentGeocacheCount(r.nextInt());
21              } else if ("MaxCacheCount".equals(name)) {
22                  builder.maxGeocacheCount(r.nextInt());
23              } else {
24                  r.skipValue();
25              }
26          }
27          r.endObject();
28  
29          return builder.build();
30      }
