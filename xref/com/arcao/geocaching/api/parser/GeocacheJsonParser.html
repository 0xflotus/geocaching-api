9   import java.util.List;

11  import static com.arcao.geocaching.api.parser.JsonParserUtil.parseAttributeList;
12  import static com.arcao.geocaching.api.parser.JsonParserUtil.parseContainerType;
13  import static com.arcao.geocaching.api.parser.JsonParserUtil.parseGeocacheType;
14  import static com.arcao.geocaching.api.parser.JsonParserUtil.parseUser;
15  
16  public final class GeocacheJsonParser {
17      private GeocacheJsonParser() {
18      }
19  
20      public static List<Geocache> parseList(JsonReader r) throws IOException {
21          if (r.peek() != JsonToken.BEGIN_ARRAY) {
22              r.skipValue();
23          }
24  
25          List<Geocache> list = new ArrayList<Geocache>();
26          r.beginArray();
27  
28          while (r.hasNext()) {
29              list.add(parse(r));
30          }
31          r.endArray();
32          return list;
33      }
34  
35      private static Geocache parse(JsonReader r) throws IOException {
36          Geocache.Builder builder = Geocache.builder();
37          Coordinates.Builder coordinatesBuilder = Coordinates.builder();
38  
39          r.beginObject();
40          while (r.hasNext()) {
41              String name = r.nextName();
42              if ("ID".equals(name)) {
43                  builder.id(r.nextLong());
44              } else if ("Code".equals(name)) {
45                  builder.code(r.nextString());
46              } else if ("Name".equals(name)) {
47                  builder.name(r.nextString());
48              } else if ("Longitude".equals(name)) {
49                  coordinatesBuilder.longitude(r.nextDouble());
50              } else if ("Latitude".equals(name)) {
51                  coordinatesBuilder.latitude(r.nextDouble());
52              } else if ("CacheType".equals(name)) {
53                  builder.geocacheType(parseGeocacheType(r));
54              } else if ("Difficulty".equals(name)) {
55                  builder.difficulty((float) r.nextDouble());
56              } else if ("Terrain".equals(name)) {
57                  builder.terrain((float) r.nextDouble());
58              } else if ("Owner".equals(name)) {
59                  builder.owner(parseUser(r));
60              } else if ("Available".equals(name)) {
61                  builder.available(r.nextBoolean());
62              } else if ("Archived".equals(name)) {
63                  builder.archived(r.nextBoolean());
64              } else if ("IsPremium".equals(name)) {
65                  builder.premium(r.nextBoolean());
66              } else if ("Country".equals(name)) {
67                  builder.countryName(r.nextString());
68              } else if ("State".equals(name)) {
69                  builder.stateName(r.nextString());
70              } else if ("DateCreated".equals(name)) {
71                  builder.createDate(JsonParserUtil.parseJsonDate(r.nextString()));
72              } else if ("PublishDateUtc".equals(name)) {
73                  builder.publishDate(JsonParserUtil.parseJsonUTCDate(r.nextString()));
74              } else if ("UTCPlaceDate".equals(name)) {
75                  builder.placeDate(JsonParserUtil.parseJsonUTCDate(r.nextString()));
76              } else if ("DateLastUpdate".equals(name)) {
77                  builder.lastUpdateDate(JsonParserUtil.parseJsonDate(r.nextString()));
78              } else if ("DateLastVisited".equals(name)) {
79                  builder.lastVisitDate(JsonParserUtil.parseJsonDate(r.nextString()));
80              } else if ("PlacedBy".equals(name)) {
81                  builder.placedBy(r.nextString());
82              } else if ("ContainerType".equals(name)) {
83                  builder.containerType(parseContainerType(r));
84              } else if ("TrackableCount".equals(name)) {
85                  builder.trackableCount(r.nextInt());
86              } else if ("HasbeenFoundbyUser".equals(name)) {
87                  builder.foundByUser(r.nextBoolean());
88              } else if ("ShortDescription".equals(name)) {
89                  builder.shortDescription(r.nextString());
90              } else if ("ShortDescriptionIsHtml".equals(name)) {
91                  builder.shortDescriptionHtml(r.nextBoolean());
92              } else if ("LongDescription".equals(name)) {
93                  builder.longDescription(r.nextString());
94              } else if ("LongDescriptionIsHtml".equals(name)) {
95                  builder.longDescriptionHtml(r.nextBoolean());
96              } else if ("EncodedHints".equals(name)) {
97                  builder.hint(r.nextString());
98              } else if ("GeocacheLogs".equals(name)) {
99                  builder.geocacheLogs(GeocacheLogJsonParser.parseList(r));
100             } else if ("Trackables".equals(name)) {
101                 builder.trackables(TrackableJsonParser.parseList(r));
102             } else if ("AdditionalWaypoints".equals(name)) {
103                 builder.waypoints(WaypointJsonParser.parseList(r));
104             } else if ("Attributes".equals(name)) {
105                 builder.attributes(parseAttributeList(r));
106             } else if ("UserWaypoints".equals(name)) {
107                 builder.userWaypoints(UserWaypointsJsonParser.parseList(r));
108             } else if ("GeocacheNote".equals(name)) {
109                 builder.personalNote(r.nextString());
110             } else if ("Images".equals(name)) {
111                 builder.images(ImageDataJsonParser.parseList(r));
112             } else if ("FavoritePoints".equals(name)) {
113                 builder.favoritePoints(r.nextInt());
114             } else if ("CanCacheBeFavorited".equals(name)) {
115                 builder.favoritable(r.nextBoolean());
116             } else if ("FoundDate".equals(name)) {
117                 builder.foundDate(JsonParserUtil.parseJsonDate(r.nextString()));
118             } else if ("HasbeenFavoritedbyUser".equals(name)) {
119                 builder.favoritedByUser(r.nextBoolean());
120             } else if ("ImageCount".equals(name)) {
121                 builder.imageCount(r.nextInt());
122             } else if ("IsRecommended".equals(name)) {
123                 builder.recommended(r.nextBoolean());
124             } else if ("TrackableCount".equals(name)) {
125                 builder.trackableCount(r.nextInt());
126             } else if ("Url".equals(name)) {
127                 builder.url(r.nextString());
128             } else if ("GUID".equals(name)) {
129                 builder.guid(r.nextString());
130             } else {
131                 r.skipValue();
132             }
133         }
134         r.endObject();
135 
136         builder.coordinates(coordinatesBuilder.build());
137 
138         return builder.build();
139     }
