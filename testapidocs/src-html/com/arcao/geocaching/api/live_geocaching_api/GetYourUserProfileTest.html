001package com.arcao.geocaching.api.live_geocaching_api;
002
003import com.arcao.geocaching.api.data.DeviceInfo;
004import com.arcao.geocaching.api.data.UserProfile;
005import com.arcao.geocaching.api.exception.GeocachingApiException;
006
007import org.junit.Assert;
008import org.junit.Test;
009
010public class GetYourUserProfileTest extends AbstractGeocachingTest {
011    private static final DeviceInfo DEVICE_INFO = DeviceInfo.builder()
012            .applicationCurrentMemoryUsage(0)
013            .applicationPeakMemoryUsage(0)
014            .applicationSoftwareVersion("1.0")
015            .deviceManufacturer("Unknown")
016            .deviceName("JUnit test")
017            .deviceOperatingSystem("N/A")
018            .deviceTotalMemoryInMb(0)
019            .deviceUniqueId("12345")
020            .build();
021
022    @Test
023    public void getYourUserProfileSimpleTest() throws GeocachingApiException {
024        UserProfile userProfile = api.getYourUserProfile(false, false, false, false, false, false, DEVICE_INFO);
025
026        Assert.assertNull(userProfile.favoritePointsStats());
027        Assert.assertNull(userProfile.geocacheFindStats());
028        Assert.assertNull(userProfile.publicProfile());
029        Assert.assertNull(userProfile.souvenirs());
030
031        Assert.assertNotNull(userProfile);
032        Assert.assertNotNull(userProfile.globalStats());
033        Assert.assertNotNull(userProfile.user());
034    }
035
036    @Test
037    public void getYourUserProfileCompleteTest() throws GeocachingApiException {
038        UserProfile userProfile = api.getYourUserProfile(true, true, true, true, false, true, DEVICE_INFO);
039
040        //Assert.assertNotNull(userProfile.getFavoritePointsStats());
041        Assert.assertNotNull(userProfile.geocacheFindStats());
042        Assert.assertNotNull(userProfile.globalStats());
043        //Assert.assertNotNull(userProfile.souvenirs());
044        //Assert.assertEquals(0, userProfile.souvenirs().size());
045
046        Assert.assertNotNull(userProfile.publicProfile());
047        Assert.assertNotNull(userProfile);
048        Assert.assertNotNull(userProfile.user());
049    }
050}
