001package com.arcao.geocaching.api.live_geocaching_api;
002
003import com.arcao.geocaching.api.data.apilimits.ApiLimits;
004import com.arcao.geocaching.api.data.apilimits.ApiLimitsResponse;
005import com.arcao.geocaching.api.data.apilimits.MaxPerPage;
006import com.arcao.geocaching.api.data.type.MemberType;
007import com.arcao.geocaching.api.exception.GeocachingApiException;
008
009import org.junit.Assert;
010import org.junit.Test;
011
012public class GetApiLimitsTest extends AbstractGeocachingTest {
013
014    @Test
015    public void GetApiLimitsCompleteTest() throws GeocachingApiException {
016        ApiLimitsResponse limitsResponse = api.getApiLimits();
017
018        Assert.assertNotNull(limitsResponse);
019
020        ApiLimits limits = limitsResponse.apiLimits();
021        Assert.assertNotNull(limits);
022        Assert.assertNotNull(limits.cacheLimits());
023        Assert.assertNotNull(limits.liteCacheLimits());
024        Assert.assertNotNull(limits.methodLimits());
025        Assert.assertEquals(MemberType.Premium, limits.forMembershipType());
026
027        MaxPerPage maxPerPage = limitsResponse.maxPerPage();
028        Assert.assertNotNull(maxPerPage);
029        Assert.assertNotSame(0, maxPerPage.geocaches());
030        Assert.assertNotSame(0, maxPerPage.geocacheLogs());
031        Assert.assertNotSame(0, maxPerPage.trackables());
032        Assert.assertNotSame(0, maxPerPage.trackableLogs());
033        Assert.assertNotSame(0, maxPerPage.cacheNotes());
034        Assert.assertNotSame(0, maxPerPage.galleryImages());
035    }
036
037}
