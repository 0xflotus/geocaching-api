036    @Test
037    public void getLiteGeocacheByCacheCodeTest() throws GeocachingApiException {
038        Geocache cache = api.getGeocache(GeocachingApi.ResultQuality.LITE, CACHE_CODE, 0, 0);
039
040        // ResultQuality.LITE
041        assertNotNull(cache);
042        assertNotNull(cache.owner());
043        assertNotSame(MemberType.Guest, cache.owner().memberType());
044        assertEquals(CACHE_CODE, cache.code());
045        assertEquals(GeocacheType.Multi, cache.geocacheType());
046        assertNotNull(cache.placedBy());
047        assertEquals(ContainerType.Micro, cache.containerType());
048        assertNotNull(cache.createDate());
049        assertNotNull(cache.placeDate());
050        assertNotNull(cache.lastUpdateDate());
051        assertEquals(1.5F, cache.difficulty(), 0);
052        assertEquals(1.5F, cache.terrain(), 0);
053        assertFalse(Double.isNaN(cache.coordinates().latitude()));
054        assertFalse(Double.isNaN(cache.coordinates().longitude()));
055        assertNotNull(cache.name());
056        assertFalse(cache.archived());
057        assertTrue(cache.available());
058        assertFalse(cache.foundByUser());
059        assertFalse(cache.premium());
060        assertNotSame(0, cache.imageCount());
061        assertNotNull(cache.url());
062        assertNotNull(cache.guid());
063
064        GeocacheLimits limits = api.getLastGeocacheLimits();
065        assertNotNull(limits);
066        assertEquals(1, api.getLastSearchResultsFound());
067    }

134        assertEquals(1.5F, cache.terrain(), 0);
135        assertFalse(Double.isNaN(cache.coordinates().latitude()));
136        assertFalse(Double.isNaN(cache.coordinates().longitude()));
137        assertNotNull(cache.name());
138        assertFalse(cache.archived());
139        assertTrue(cache.available());
140        assertFalse(cache.foundByUser());
141        assertFalse(cache.premium());
142        assertNotSame(0, cache.imageCount());
143        assertNotNull(cache.url());
144        assertNotNull(cache.guid());

146
147        // ResultQuality.SUMMARY
148        List<Waypoint> waypoints = cache.waypoints();
149        assertNotNull(waypoints);
150        assertEquals(8, waypoints.size());
151        assertNotNull(cache.hint());
152        assertNotNull(cache.shortDescription());
153        assertTrue(cache.shortDescriptionHtml());

157        assertNotNull(images);
158        assertFalse(images.isEmpty());
159
160        // ResultQuality.FULL
161        assertNotNull(cache.countryName());
162        assertNotNull(cache.stateName());
163
164        List<GeocacheLog> geocacheLogs = cache.geocacheLogs();
165        assertNotNull(geocacheLogs);
166        assertEquals(5, geocacheLogs.size());
167        for (GeocacheLog geocacheLog : geocacheLogs) {
168            assertNotNull(geocacheLog.author());
169            assertNotNull(geocacheLog.created());
170            assertNotNull(geocacheLog.visited());
171            assertNotSame(GeocacheLogType.Unknown, geocacheLog.logType());
172            assertNotNull(geocacheLog.text());
173        }
174
175        EnumSet<AttributeType> attributes = cache.attributes();
176        assertNotNull(attributes);
177        assertFalse(attributes.isEmpty());
178        assertNotNull(cache.countryName());
179        assertNotNull(cache.stateName());
