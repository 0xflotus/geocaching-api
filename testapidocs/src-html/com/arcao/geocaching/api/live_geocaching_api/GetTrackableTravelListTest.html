001package com.arcao.geocaching.api.live_geocaching_api;
002
003import com.arcao.geocaching.api.data.TrackableTravel;
004import com.arcao.geocaching.api.exception.GeocachingApiException;
005
006import org.junit.Assert;
007import org.junit.Test;
008
009import java.util.List;
010
011
012public class GetTrackableTravelListTest extends AbstractGeocachingTest {
013        private static final String TRACKABLE_CODE = "TB3GW4D";
014        private static final double TRACKING_HISTORY = 144.03; // source: http://staging.geocaching.com/track/details.aspx?id=2862183
015        
016        @Test
017        public void simpleGetTrackableTravelListTest() throws GeocachingApiException {
018                List<TrackableTravel> travel = api.getTrackableTravelList(TRACKABLE_CODE);
019                
020                TrackableTravel previousPoint = null;
021                double history = 0D;
022                
023                for(TrackableTravel point : travel) {
024                        if (previousPoint != null) {
025                                history += previousPoint.coordinates().distanceTo(point.coordinates());
026                        }
027                        
028                        previousPoint = point;
029                }
030                
031                Assert.assertEquals(TRACKING_HISTORY, history / 1000D, 0.2);
032        }
033}
