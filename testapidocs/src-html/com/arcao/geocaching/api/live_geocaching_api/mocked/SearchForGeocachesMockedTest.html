027
028public class SearchForGeocachesMockedTest extends AbstractMockedGeocachingTest {
029    private final static String CACHE_CODE = "GCY81P";
030
031    @Test
032    public void getLiteGeocacheByCacheCodeNotFoundTest() throws GeocachingApiException {
033        JsonReader resultBody = createJsonReaderFromResource("result-lite-geocache-by-cache-code-not-found.json");
034
035        when(downloader.post(any(URL.class), any(byte[].class)))
036                .thenReturn(resultBody);
037
038        Geocache geocache = api.getGeocache(GeocachingApi.ResultQuality.LITE, CACHE_CODE, 0, 0);
039        assertNull(geocache);
040    }
041
042    @Test
043    public void getLiteGeocacheByCacheCodeRequestTest() throws GeocachingApiException, IOException {
044        byte[] expectedRequestBody = createByteArrayFromResource("request-lite-geocache-by-cache-code.json");
045        JsonReader resultBody = createJsonReaderFromResource("result-lite-geocache-by-cache-code.json");
046
047        when(downloader.post(any(URL.class), any(byte[].class)))
048                .thenReturn(resultBody);
049
050        Geocache geocache = api.getGeocache(GeocachingApi.ResultQuality.LITE, CACHE_CODE, 0, 0);
051
052        verify(downloader).post(any(URL.class), eq(expectedRequestBody));
053
054        // ResultQuality.LITE
055        assertNotNull(geocache);
056        assertNotNull(geocache.owner());
057        assertNotSame(MemberType.Guest, geocache.owner().memberType());
058        assertEquals(CACHE_CODE, geocache.code());
059        assertEquals(GeocacheType.Multi, geocache.geocacheType());
060        assertNotNull(geocache.placedBy());
061        assertEquals(ContainerType.Micro, geocache.containerType());
062        assertNotNull(geocache.createDate());
063        assertNotNull(geocache.placeDate());
064        assertNotNull(geocache.lastUpdateDate());
065        assertEquals(1.5F, geocache.difficulty(), 0);
066        assertEquals(1.5F, geocache.terrain(), 0);
067        assertFalse(Double.isNaN(geocache.coordinates().latitude()));
068        assertFalse(Double.isNaN(geocache.coordinates().longitude()));
069        assertNotNull(geocache.name());
070        assertFalse(geocache.archived());
071        assertTrue(geocache.available());
072        assertFalse(geocache.foundByUser());
073        assertFalse(geocache.premium());
074        assertNotSame(0, geocache.imageCount());
075        assertNotNull(geocache.url());
076        assertNotNull(geocache.guid());
077
078        GeocacheLimits limits = api.getLastGeocacheLimits();
079        assertNotNull(limits);
080        assertEquals(1, api.getLastSearchResultsFound());
081
082        GeocacheLimits geocacheLimits = api.getLastGeocacheLimits();
083        assertNotNull(geocacheLimits);
084        assertEquals(131, geocacheLimits.currentGeocacheCount());
085        assertEquals(9869, geocacheLimits.geocacheLeft());
086        assertEquals(10000, geocacheLimits.maxGeocacheCount());
087    }
