001package com.arcao.geocaching.api.live_geocaching_api.mocked;
002
003import com.arcao.geocaching.api.GeocachingApi;
004import com.arcao.geocaching.api.LiveGeocachingApi;
005import com.arcao.geocaching.api.downloader.JsonDownloader;
006import com.arcao.geocaching.api.exception.GeocachingApiException;
007import com.arcao.geocaching.api.parser.JsonReader;
008
009import org.junit.Before;
010
011import java.io.ByteArrayOutputStream;
012import java.io.IOException;
013import java.io.InputStream;
014import java.io.InputStreamReader;
015import java.nio.charset.Charset;
016
017import static org.mockito.Mockito.mock;
018
019@SuppressWarnings("WeakerAccess")
020public abstract class AbstractMockedGeocachingTest {
021    private static final Charset CHARSET_UTF8 = Charset.forName("UTF-8");
022
023    GeocachingApi api = null;
024    JsonDownloader downloader = null;
025
026    private static final String ACCESS_TOKEN = "ACCESS_TOKEN";
027
028    @Before
029    public void setUpMock() throws GeocachingApiException {
030        downloader = mock(JsonDownloader.class);
031
032        api = LiveGeocachingApi.builder()
033                .downloader(downloader)
034                .build();
035
036        api.openSession(ACCESS_TOKEN);
037    }
038
039    JsonReader createJsonReaderFromResource(String resourceFile) {
040        return new JsonReader(new InputStreamReader(getClass().getResourceAsStream(resourceFile), CHARSET_UTF8));
041    }
042
043    byte[] createByteArrayFromResource(String resourceFile) throws IOException {
044        InputStream inputStream = getClass().getResourceAsStream(resourceFile);
045        ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
046
047        byte[] buffer = new byte[4096];
048
049        try {
050            while (true) {
051                int len = inputStream.read(buffer);
052                if (len < 0)
053                    break;
054
055                outputStream.write(buffer, 0, len);
056            }
057        } finally {
058            inputStream.close();
059        }
060
061        return outputStream.toByteArray();
062
063    }
064}
