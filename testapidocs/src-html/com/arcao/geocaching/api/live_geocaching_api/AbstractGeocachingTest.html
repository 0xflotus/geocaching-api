001package com.arcao.geocaching.api.live_geocaching_api;
002
003import com.arcao.geocaching.api.GeocachingApi;
004import com.arcao.geocaching.api.LiveGeocachingApi;
005import com.arcao.geocaching.api.configuration.GeocachingApiConfiguration;
006import com.arcao.geocaching.api.downloader.DefaultJsonDownloader;
007import com.arcao.geocaching.api.downloader.JsonDownloader;
008import com.arcao.geocaching.api.exception.GeocachingApiException;
009
010import org.apache.commons.lang3.StringUtils;
011import org.junit.BeforeClass;
012
013public abstract class AbstractGeocachingTest {
014    protected static GeocachingApi api = null;
015
016    // generated via oAuth example, see: https://github.com/arcao/geocaching-api-examples
017    private static final String STAGING_AUTH_TOKEN = "JvJcmhBJ88iDoWR107RIKpQgLLU=";
018
019    @BeforeClass
020    public static void setUp() throws GeocachingApiException {
021        LiveGeocachingApi.Builder builder = LiveGeocachingApi.builder();
022
023        GeocachingApiConfiguration configuration = GeocachingApiConfiguration.STAGING;
024
025        if (!Boolean.parseBoolean(StringUtils.defaultIfEmpty(System.getenv("GEOCACHING_STAGING"), "true"))) {
026            configuration = GeocachingApiConfiguration.PRODUCTION;
027        }
028
029        JsonDownloader downloader = new DefaultJsonDownloader(configuration).debug(true);
030
031        builder.configuration(configuration);
032        builder.downloader(downloader);
033
034        api = builder.build();
035
036        api.openSession(StringUtils.defaultIfEmpty(System.getenv("GEOCACHING_TOKEN"), STAGING_AUTH_TOKEN));
037    }
038}
