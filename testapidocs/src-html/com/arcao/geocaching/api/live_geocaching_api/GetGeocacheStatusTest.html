001package com.arcao.geocaching.api.live_geocaching_api;
002
003import com.arcao.geocaching.api.data.GeocacheStatus;
004import com.arcao.geocaching.api.data.type.GeocacheType;
005import com.arcao.geocaching.api.exception.GeocachingApiException;
006import com.arcao.geocaching.api.exception.LiveGeocachingApiException;
007
008import org.junit.Assert;
009import org.junit.Test;
010
011import java.util.Arrays;
012import java.util.Collections;
013import java.util.List;
014
015public class GetGeocacheStatusTest extends AbstractGeocachingTest {
016    private static final List<String> ZERO_CACHE_CODES = Collections.emptyList();
017    private static final List<String> ONE_CACHE_CODE = Collections.singletonList("GCXZE8");
018    private static final List<String> TWO_CACHE_CODES = Arrays.asList("GCXZE8", "GCY81P");
019
020    @Test(expected = LiveGeocachingApiException.class)
021    public void zeroGetGeocacheStatusTest() throws GeocachingApiException {
022        api.getGeocacheStatus(ZERO_CACHE_CODES);
023
024        Assert.fail();
025    }
026
027    @Test
028    public void oneGetGeocacheStatusTest() throws GeocachingApiException {
029        List<GeocacheStatus> geocacheStatusList = api.getGeocacheStatus(ONE_CACHE_CODE);
030
031        Assert.assertNotNull(geocacheStatusList);
032        Assert.assertEquals(1, geocacheStatusList.size());
033
034        GeocacheStatus status = geocacheStatusList.get(0);
035        Assert.assertEquals(ONE_CACHE_CODE.get(0), status.cacheCode());
036        Assert.assertEquals("TB Hotel Skalni mlyn", status.cacheName());
037        Assert.assertEquals(GeocacheType.Mystery, status.cacheType());
038
039        Assert.assertFalse(status.archived());
040        Assert.assertTrue(status.available());
041        Assert.assertFalse(status.premium());
042
043        Assert.assertNotSame(0, status.trackableCount());
044    }
045
046    @Test
047    public void twoGetGeocacheStatusTest() throws GeocachingApiException {
048        List<GeocacheStatus> geocacheStatusList = api.getGeocacheStatus(TWO_CACHE_CODES);
049
050        Assert.assertNotNull(geocacheStatusList);
051        Assert.assertEquals(2, geocacheStatusList.size());
052    }
053
054}
