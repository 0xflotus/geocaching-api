001package com.arcao.geocaching.api.util;
002
003import org.junit.Test;
004
005import static org.junit.Assert.*;
006
007public class GeocachingUtilsTest {
008  @Test
009  public void testBase31Decode() {
010    assertEquals(1,     GeocachingUtils.base31Decode("1"));
011    assertEquals(31,    GeocachingUtils.base31Decode("10"));
012    assertEquals(31*31, GeocachingUtils.base31Decode("100"));
013  }
014  
015  @Test
016  public void testBase31Encode() {
017    assertEquals("1",   GeocachingUtils.base31Encode(1));
018    assertEquals("10",  GeocachingUtils.base31Encode(31));
019    assertEquals("100", GeocachingUtils.base31Encode(31*31));
020  }
021
022  @Test
023  public void testCacheCodeToCacheId() {
024    assertEquals(0x0,     GeocachingUtils.cacheCodeToCacheId("GC0"));
025    assertEquals(0xFFFF,  GeocachingUtils.cacheCodeToCacheId("GCFFFF"));
026    assertEquals(0x10000, GeocachingUtils.cacheCodeToCacheId("GCG000"));
027    assertEquals(95327,   GeocachingUtils.cacheCodeToCacheId("GCH000"));
028    assertEquals(1272588, GeocachingUtils.cacheCodeToCacheId("GC1TG15"));
029  }
030
031  @Test(expected = IllegalArgumentException.class)
032  public void testInvalidCacheCodeToCacheId1() {
033    GeocachingUtils.cacheCodeToCacheId("GC*-/");
034    fail();
035  }
036
037  @Test(expected = IllegalArgumentException.class)
038  public void testInvalidCacheCodeToCacheId2() {
039    GeocachingUtils.cacheCodeToCacheId("GCIL");
040    fail();
041  }
042
043
044  @Test
045  public void testCacheIdToCacheCode() {
046    assertEquals("GC0",     GeocachingUtils.cacheIdToCacheCode(0));
047    assertEquals("GCFFFF",  GeocachingUtils.cacheIdToCacheCode(0xFFFF));
048    assertEquals("GCG000",  GeocachingUtils.cacheIdToCacheCode(0x10000));
049    assertEquals("GCH000",  GeocachingUtils.cacheIdToCacheCode(95327));
050    assertEquals("GC1TG15", GeocachingUtils.cacheIdToCacheCode(1272588));
051  }
052
053  @Test
054  public void testIsCacheCodeValid() {
055    assertTrue(GeocachingUtils.isCacheCodeValid("GC0"));
056    assertTrue(GeocachingUtils.isCacheCodeValid("GCFFFF"));
057    assertTrue(GeocachingUtils.isCacheCodeValid("GCG000"));
058    assertTrue(GeocachingUtils.isCacheCodeValid("GCH000"));
059    assertTrue(GeocachingUtils.isCacheCodeValid("GC1TG15"));
060    assertFalse(GeocachingUtils.isCacheCodeValid("GC*-/"));
061    assertFalse(GeocachingUtils.isCacheCodeValid("GCIL"));
062    assertFalse(GeocachingUtils.isCacheCodeValid("GC"));
063    assertFalse(GeocachingUtils.isCacheCodeValid("OC1234"));
064  }
065}
