001package com.arcao.geocaching.api.data.coordinates;
002
003import org.junit.Test;
004
005import java.text.DecimalFormatSymbols;
006import java.text.ParseException;
007
008import static org.junit.Assert.assertEquals;
009
010
011public class CoordinatesFormatterTest {
012  private static final double refLatitude = 49.0 + 56.031 / 60.0;
013  private static final double refLongitude = 8.0 + 38.564 / 60.0;
014  private static final Coordinates refCoordinates = Coordinates.create(refLatitude, refLongitude);
015  private static final char decSymbol = new DecimalFormatSymbols().getDecimalSeparator(); // some tests are Locale specific!!!
016  
017  @Test
018  public void testLatLonDecDegree() throws ParseException {
019    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_LON_DECDEGREE);
020    assertEquals("49" + decSymbol + "933850 8" + decSymbol + "642733", f.format(refCoordinates));
021  }
022  
023  @Test
024  public void testLatLonDecDegreeComma() throws ParseException {
025    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_LON_DECDEGREE_COMMA);
026    assertEquals("49.933850,8.642733", f.format(refCoordinates));
027  }
028  
029  @Test
030  public void testLatLonDecMinute() throws ParseException {
031    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_LON_DECMINUTE);
032    assertEquals("N 49° 56" + decSymbol + "031 · E 8° 38" + decSymbol + "564", f.format(refCoordinates));
033  }
034  
035  @Test
036  public void testLatLonDecMinuteRaw() throws ParseException {
037    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_LON_DECMINUTE_RAW);
038    assertEquals("N 49° 56.031 E 8° 38.564", f.format(refCoordinates));
039  }
040  
041  @Test
042  public void testLatLonDecSecond() throws ParseException {
043    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_LON_DECSECOND);
044    assertEquals("N 49° 56' 01" + decSymbol + "860\" · E 8° 38' 33" + decSymbol + "840\"", f.format(refCoordinates));
045  }
046  
047  @Test
048  public void testLatDecDegreeRaw() throws ParseException {
049    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_DECDEGREE_RAW);
050    assertEquals("49.933850", f.format(refCoordinates));
051  }
052  
053  @Test
054  public void testLatDecMinute() throws ParseException {
055    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_DECMINUTE);
056    assertEquals("N 49° 56" + decSymbol + "031", f.format(refCoordinates));
057  }
058  
059  @Test
060  public void testLatDecMinuteRaw() throws ParseException {
061    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_DECMINUTE_RAW);
062    assertEquals("N 49 56" + decSymbol + "031", f.format(refCoordinates));
063  }
064  
065  @Test
066  public void testLonDecDegreeRaw() throws ParseException {
067    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LON_DECDEGREE_RAW);
068    assertEquals("8.642733", f.format(refCoordinates));
069  }
070  
071  @Test
072  public void testLonDecMinute() throws ParseException {
073    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LON_DECMINUTE);
074    assertEquals("E 8° 38" + decSymbol + "564", f.format(refCoordinates));
075  }
076  
077  @Test
078  public void testLonDecMinuteRaw() throws ParseException {
079    CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LON_DECMINUTE_RAW);
080    assertEquals("E 8 38" + decSymbol + "564", f.format(refCoordinates));
081  }
082}
