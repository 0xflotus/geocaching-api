001package com.arcao.geocaching.api.data.coordinates;
002
003import org.junit.Test;
004
005import java.text.ParseException;
006
007import static org.junit.Assert.assertEquals;
008import static org.junit.Assert.assertNotNull;
009import static org.junit.Assert.assertTrue;
010
011public class CoordinatesParserTest {
012  private static final double refLongitude = 8.0 + 38.564 / 60.0;
013  private static final double refLatitude = 49.0 + 56.031 / 60.0;
014  private static final Coordinates refCoordinates = Coordinates.create(refLatitude, refLongitude);
015  private static final double PRECISION = 1e-8;
016
017  @Test
018  public void testParseLatitude() throws ParseException {
019    assertEquals(refLatitude, CoordinatesParser.parseLatitude("N 49° 56.031"), PRECISION);
020  }
021
022  @Test
023  public void testParseLongitude() throws ParseException {
024    assertEquals(refLongitude, CoordinatesParser.parseLongitude("E 8° 38.564"), PRECISION);
025  }
026
027  @Test
028  public void testFullCoordinates() throws ParseException {
029    assertTrue(refCoordinates.equals(Coordinates.parseCoordinates("N 49° 56.031 | E 8° 38.564")));
030  }
031
032  @Test
033  public void testCoordinateMissingPart() {
034    // we are trying to parse a _point_, but have only a latitude. Don't accept
035    // the numerical part as longitude!
036    Coordinates point = null;
037    try {
038      point = CoordinatesParser.parse("N 49° 56.031");
039    } catch (ParseException e) {
040      // expected
041    }
042    assertEquals(null, point);
043  }
044
045  @Test
046  public void testSouth() throws ParseException {
047    assertEquals(-refLatitude, CoordinatesParser.parseLatitude("S 49° 56.031"), PRECISION);
048  }
049
050  @Test
051  public void testWest() throws ParseException {
052    assertEquals(-refLongitude, CoordinatesParser.parseLongitude("W 8° 38.564"), PRECISION);
053  }
054
055  @Test
056  public void testLowerCase() throws ParseException {
057    assertEquals(refLongitude, CoordinatesParser.parseLongitude("e 8° 38.564"), PRECISION);
058  }
059
060  @Test
061  public void testVariousFormats() throws ParseException {
062    final Coordinates point1 = CoordinatesParser.parse("N 49° 43' 57\" | E 2 12' 35");
063    final Coordinates point2 = CoordinatesParser.parse("N 49 43.95 E2°12.5833333333");
064    assertTrue(point1.equals(point2));
065  }
066
067  @Test
068  public void testInSentence() throws ParseException {
069    final Coordinates p1 = CoordinatesParser.parse("Station3: N51 21.523 / E07 02.680");
070    final Coordinates p2 = CoordinatesParser.parse("N51 21.523", "E07 02.680");
071    assertNotNull(p1);
072    assertNotNull(p2);
073    assertTrue(p1.equals(p2));
074  }
075
076  @Test
077  public void testUnrelatedParts() {
078    Coordinates point = null;
079    try {
080      point = CoordinatesParser.parse("N51 21.523 and some words in between, so there is no relation E07 02.680");
081    } catch (ParseException e) {
082      // expected
083    }
084    assertEquals(null, point);
085  }
086
087  @Test
088  public void testComma() throws ParseException {
089    final Coordinates pointComma = CoordinatesParser.parse("N 46° 27' 55,65''\n" +
090        "E 15° 53' 41,68''");
091    final Coordinates pointDot = CoordinatesParser.parse("N 46° 27' 55.65''\n" +
092        "E 15° 53' 41.68''");
093    assertNotNull(pointComma);
094    assertNotNull(pointDot);
095    assertTrue(pointComma.equals(pointDot));
096  }
097  
098  @Test
099  public void testWithoutCardinalPoint() throws ParseException {
100    assertEquals(refLatitude, CoordinatesParser.parseLatitude("49° 56.031", false), PRECISION);
101    assertEquals(refLongitude, CoordinatesParser.parseLongitude("8° 38.564", false), PRECISION);
102    
103    assertEquals(-refLatitude, CoordinatesParser.parseLatitude("-49° 56.031", false), PRECISION);
104    assertEquals(-refLongitude, CoordinatesParser.parseLongitude("-8° 38.564", false), PRECISION);
105  }
106
107  @Test
108  public void testParseTooBigCoordinatesNumber() throws ParseException {
109    Coordinates point = CoordinatesParser.parse("N 123456789123456789 | N 123456789123456789");
110    assertNotNull(point);
111  }
112
113  @Test
114  public void testParseTooBigLatitudeNumber() throws ParseException {
115    CoordinatesParser.parseLatitude("N 123456789123456789");
116    assertTrue(true);
117  }
118
119  @Test
120  public void testParseTooBigLongitudeNumber() throws ParseException {
121    CoordinatesParser.parseLongitude("W 123456789123456789");
122    assertTrue(true);
123  }
124
125  @Test
126  public void testParseCoordinatesNumberAndNormalize() throws ParseException {
127    assertTrue(refCoordinates.equals(CoordinatesParser.parse("N 229° 56.031 | E 368° 38.564")));
128  }
129}
