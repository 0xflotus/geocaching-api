001package com.arcao.geocaching.api.downloader;
002
003import com.arcao.geocaching.api.exception.InvalidResponseException;
004import com.arcao.geocaching.api.exception.NetworkException;
005import com.arcao.geocaching.api.parser.JsonReader;
006
007import java.net.URL;
008
009/**
010 * Interface to implement downloading JSON data with GET and POST method
011 *
012 * @author arcao
013 */
014public interface JsonDownloader {
015    /**
016     * Send request to a specified url with GET method and returns instance of
017     * {@link JsonReader} with a response
018     *
019     * @param url url to request including query parameters
020     * @return instance of {@link JsonReader} object with a response
021     * @throws NetworkException         If Network I/O error occurs
022     * @throws InvalidResponseException If response contains invalid data or HTTP response code isn't 200 (OK)
023     */
024    JsonReader get(URL url) throws NetworkException, InvalidResponseException;
025
026    /**
027     * Send request to a specified url with POST method including post data and returns
028     * instance of {@link JsonReader} with a response
029     *
030     * @param url      url to request including query parameters
031     * @param postData post data
032     * @return instance of {@link JsonReader} object with a response
033     * @throws NetworkException         If Network I/O error occurs
034     * @throws InvalidResponseException If response contains invalid data or HTTP response code isn't 200 (OK)
035     */
036    JsonReader post(URL url, byte[] postData) throws NetworkException, InvalidResponseException;
037}
