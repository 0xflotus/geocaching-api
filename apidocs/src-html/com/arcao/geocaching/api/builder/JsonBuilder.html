001package com.arcao.geocaching.api.builder;
002
003import org.jetbrains.annotations.Nullable;
004
005import java.util.Date;
006import java.util.Locale;
007
008/**
009 * Helper class during JSON serializing process
010 *
011 * @author arcao
012 */
013public final class JsonBuilder {
014    private JsonBuilder() {
015    }
016
017    /**
018     * Convert Date object to JSON date notation
019     *
020     * @param date date convert
021     * @return JSON notation for date
022     */
023    public static String dateToJsonString(@Nullable Date date) {
024        if (date == null)
025            return null;
026
027        return String.format(Locale.US, "/Date(%d)/", date.getTime());
028    }
029}
