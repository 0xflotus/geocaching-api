001package com.arcao.geocaching.api;
002
003import com.arcao.geocaching.api.data.FieldNote;
004import com.arcao.geocaching.api.data.Geocache;
005import com.arcao.geocaching.api.data.GeocacheLog;
006import com.arcao.geocaching.api.data.ImageData;
007import com.arcao.geocaching.api.exception.GeocachingApiException;
008import com.arcao.geocaching.api.filter.CacheCodeFilter;
009import com.arcao.geocaching.api.filter.Filter;
010
011import org.jetbrains.annotations.NotNull;
012import org.jetbrains.annotations.Nullable;
013
014import java.util.Collections;
015import java.util.List;
016
017/**
018 * Abstract class of Geocaching API
019 *
020 * @author arcao
021 * @since 1.4.3
022 */
023abstract class AbstractGeocachingApi implements GeocachingApi {
024    @Nullable String session;
025
026    @Nullable
027    @Override
028    public String getSession() {
029        return session;
030    }
031
032    @Override
033    public void openSession(@NotNull String session) throws GeocachingApiException {
034        this.session = session;
035    }
036
037    @Override
038    public Geocache getGeocache(@NotNull ResultQuality resultQuality, @NotNull String cacheCode, int cacheLogCount, int trackableLogCount) throws GeocachingApiException {
039        final List<Geocache> list = searchForGeocaches(resultQuality, 1, cacheLogCount, trackableLogCount,
040                Collections.singletonList((Filter) new CacheCodeFilter(cacheCode)), null);
041
042        if (list.isEmpty())
043            return null;
044
045        return list.get(0);
046    }
047
048    @Override
049    public GeocacheLog createFieldNoteAndPublish(@NotNull FieldNote fieldNote, boolean publish, ImageData imageData, boolean favoriteThisCache) throws GeocachingApiException {
050        return createFieldNoteAndPublish(fieldNote.cacheCode(), fieldNote.logType(), fieldNote.dateLogged(), fieldNote.note(), publish, imageData,
051                favoriteThisCache);
052    }
053}
