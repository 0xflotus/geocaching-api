022public final class JsonParserUtil {
023    private static final Logger logger = LoggerFactory.getLogger(JsonParserUtil.class);
024    private static final long HOUR_IN_MS = 1000 * 60 * 60;

026    private JsonParserUtil() {

029    static Date parseJsonDate(String date) {
030        Pattern DATE_PATTERN = Pattern.compile("/Date\\((-?\\d+)([-+]\\d{4})?\\)/");
031
032        if (date == null)
033            return null;
034
035        Matcher m = DATE_PATTERN.matcher(date);
036        if (m.matches()) {
037            long time = Long.parseLong(m.group(1));
038            long zone = 0;
039            if (m.group(2) != null && !m.group(2).isEmpty())
040                zone = Integer.parseInt(m.group(2)) / 100 * HOUR_IN_MS;
041            return new Date(time + zone);
042        }
043
044        logger.error("parseJsonDate failed: " + date);
045        return null;
046    }
047
048    static Date parseJsonUTCDate(String date) {
049        Pattern DATE_PATTERN = Pattern.compile("/Date\\((-?\\d+)([-+]\\d{4})?\\)/");
050
051        if (date == null)
052            return null;
053
054        Matcher m = DATE_PATTERN.matcher(date);
055        if (m.matches()) {
056            long time = Long.parseLong(m.group(1));
057            // zone is always zero for UTC
058            long zone = 0;
059            return new Date(time + zone);
060        }
061
062        logger.error("parseJsonDate failed: " + date);
063        return null;
064    }

066    static GeocacheType parseGeocacheType(JsonReader r) throws IOException {
067        GeocacheType geocacheType = null;
068
069        if (isNextNull(r))
070            return null;
071
072        r.beginObject();
073        while (r.hasNext()) {
074            String name = r.nextName();
075            if ("GeocacheTypeId".equals(name)) {
076                geocacheType = GeocacheType.fromId(r.nextInt());
077            } else {
078                r.skipValue();
079            }
080        }
081        r.endObject();
082        return geocacheType;
083    }
084
085    static ContainerType parseContainerType(JsonReader r) throws IOException {
086        ContainerType containerType = ContainerType.NotChosen;
087
088        if (isNextNull(r))
089            return containerType;
090
091        r.beginObject();
092        while (r.hasNext()) {
093            String name = r.nextName();
094            if ("ContainerTypeId".equals(name)) {
095                containerType = ContainerType.fromId(r.nextInt());
096            } else {
097                r.skipValue();
098            }
099        }
100        r.endObject();
101        return containerType;
102    }
103
104    static MemberType parseMemberType(JsonReader r) throws IOException {
105        MemberType memberType = null;
106
107        if (isNextNull(r))
108            return null;
109
110        if (r.peek() == JsonToken.NUMBER) {
111            memberType = MemberType.fromId(r.nextInt() / 10);
112            return memberType;
113        }
114
115        r.beginObject();
116        while (r.hasNext()) {
117            String name = r.nextName();
118            if ("MemberTypeId".equals(name)) {
119                memberType = MemberType.fromId(r.nextInt());
120            } else {
121                r.skipValue();
122            }
123        }
124        r.endObject();
125        return memberType;
126    }
127
128    private static Coordinates parseHomeCoordinates(JsonReader r) throws IOException {
129        if (isNextNull(r))
130            return null;
131
132        Coordinates.Builder coordinates = Coordinates.builder();
133
134        r.beginObject();
135        while (r.hasNext()) {
136            String name = r.nextName();
137            if ("Latitude".equals(name)) {
138                coordinates.latitude((float) r.nextDouble());
139            } else if ("Longitude".equals(name)) {
140                coordinates.longitude((float) r.nextDouble());
141            } else {
142                r.skipValue();
143            }
144        }
145        r.endObject();
146        return coordinates.build();
147    }
148
149    static User parseUser(JsonReader r) throws IOException {
150        if (isNextNull(r))
151            return null;
152
153        User.Builder user = User.builder();
154        String userName = null;
155
156        r.beginObject();
157        while (r.hasNext()) {
158            String name = r.nextName();
159            if ("AvatarUrl".equals(name)) {
160                user.avatarUrl(r.nextString());
161            } else if ("FindCount".equals(name)) {
162                user.findCount(r.nextInt());
163            } else if ("HideCount".equals(name)) {
164                user.hideCount(r.nextInt());
165            } else if ("HomeCoordinates".equals(name)) {
166                user.homeCoordinates(parseHomeCoordinates(r));
167            } else if ("Id".equals(name)) {
168                user.id(r.nextLong());
169            } else if ("IsAdmin".equals(name)) {
170                user.admin(r.nextBoolean());
171            } else if ("MemberType".equals(name)) {
172                user.memberType(parseMemberType(r));
173            } else if ("PublicGuid".equals(name)) {
174                user.publicGuid(r.nextString());
175            } else if ("UserName".equals(name)) {
176                userName = r.nextString();
177                user.userName(userName);
178            } else {
179                r.skipValue();
180            }
181        }
182        r.endObject();
183
184        if (userName == null)
185            return null;
186
187        return user.build();
188    }
189
190    private static AttributeType parseAttribute(JsonReader r) throws IOException {
191        int id = 1;
192        boolean on = false;
193
194        r.beginObject();
195        while (r.hasNext()) {
196            String name = r.nextName();
197            if ("AttributeTypeID".equals(name)) {
198                id = r.nextInt();
199            } else if ("IsOn".equals(name)) {
200                on = r.nextBoolean();
201            } else {
202                r.skipValue();
203            }
204        }
205        r.endObject();
206
207        return AttributeType.fromId(id, on);
208    }
209
210    static EnumSet<AttributeType> parseAttributeList(JsonReader r) throws IOException {
211        if (r.peek() != JsonToken.BEGIN_ARRAY) {
212            r.skipValue();
213        }
214
215        EnumSet<AttributeType> attributeSet = EnumSet.noneOf(AttributeType.class);
216        r.beginArray();
217        while (r.hasNext()) {
218            AttributeType attribute = parseAttribute(r);
219            if (attribute != null) {
220                attributeSet.add(attribute);
221            }
222        }
223        r.endArray();
224        return attributeSet;
225    }

236            if ("WptLogTypeId".equals(name)) {
237                trackableLogType = TrackableLogType.fromId(r.nextInt());
238            } else {
239                r.skipValue();
240            }
241        }
242        r.endObject();
243        return trackableLogType;
244    }
245
246    static boolean isNextNull(JsonReader r) throws IOException {
247        if (r.peek() == JsonToken.NULL) {
248            r.nextNull();
249            return true;
250        }
251
252        return false;
253    }
