001package com.arcao.geocaching.api.parser;
002
003import com.arcao.geocaching.api.data.UserProfile;
004import com.arcao.geocaching.api.data.userprofile.FavoritePointStats;
005import com.arcao.geocaching.api.data.userprofile.GeocacheFindStats;
006import com.arcao.geocaching.api.data.userprofile.GlobalStats;
007import com.arcao.geocaching.api.data.userprofile.ProfilePhoto;
008import com.arcao.geocaching.api.data.userprofile.PublicProfile;
009import com.arcao.geocaching.api.data.userprofile.TrackableStats;
010
011import java.io.IOException;
012
013import static com.arcao.geocaching.api.parser.JsonParserUtil.isNextNull;
014import static com.arcao.geocaching.api.parser.JsonParserUtil.parseJsonDate;
015import static com.arcao.geocaching.api.parser.JsonParserUtil.parseUser;
016import static com.arcao.geocaching.api.parser.TrackableJsonParser.parseList;
017
018public final class UserProfileJsonParser {
019    private UserProfileJsonParser() {
020    }
021
022    public static UserProfile parse(JsonReader r) throws IOException {
023        if (isNextNull(r))
024            return null;
025
026        UserProfile.Builder builder = UserProfile.builder();
027
028        r.beginObject();
029        while (r.hasNext()) {
030            String name = r.nextName();
031            if ("FavoritePoints".equals(name)) {
032                builder.favoritePointsStats(parseFavoritePointStats(r));
033            } else if ("Geocaches".equals(name)) {
034                builder.geocacheFindStats(parseGeocacheFindStats(r));
035            } else if ("PublicProfile".equals(name)) {
036                builder.publicProfile(parsePublicProfile(r));
037            } else if ("Stats".equals(name)) {
038                builder.globalStats(parseGlobalStats(r));
039            } else if ("Trackables".equals(name)) {
040                builder.trackableStats(parseTrackableStats(r));
041            } else if ("User".equals(name)) {
042                builder.user(parseUser(r));
043            } else {
044                r.skipValue();
045            }
046        }
047        r.endObject();
048
049        return builder.build();
050    }
051
052    private static FavoritePointStats parseFavoritePointStats(JsonReader r) throws IOException {
053        if (isNextNull(r))
054            return null;
055
056        FavoritePointStats.Builder builder = FavoritePointStats.builder();
057
058        r.beginObject();
059        while (r.hasNext()) {
060            r.nextName();
061            r.skipValue();
062        }
063        r.endObject();
064
065        return builder.build();
066    }
067
068    private static GeocacheFindStats parseGeocacheFindStats(JsonReader r) throws IOException {
069        if (isNextNull(r))
070            return null;
071
072        GeocacheFindStats.Builder builder = GeocacheFindStats.builder();
073
074        r.beginObject();
075        while (r.hasNext()) {
076            r.nextName();
077            r.skipValue();
078        }
079        r.endObject();
080
081        return builder.build();
082    }
083
084    private static PublicProfile parsePublicProfile(JsonReader r) throws IOException {
085        if (isNextNull(r))
086            return null;
087
088        PublicProfile.Builder builder = PublicProfile.builder();
089
090        r.beginObject();
091        while (r.hasNext()) {
092            String name = r.nextName();
093            if ("ForumTitle".equals(name)) {
094                builder.forumTitle(r.nextString());
095            } else if ("LastVisit".equals(name)) {
096                builder.lastVisit(parseJsonDate(r.nextString()));
097            } else if ("Location".equals(name)) {
098                builder.location(r.nextString());
099            } else if ("MemberSince".equals(name)) {
100                builder.memberSince(parseJsonDate(r.nextString()));
101            } else if ("Occupation".equals(name)) {
102                builder.occupation(r.nextString());
103            } else if ("ProfilePhoto".equals(name)) {
104                builder.profilePhoto(parseProfilePhoto(r));
105            } else if ("ProfileText".equals(name)) {
106                builder.profileText(r.nextString());
107            } else {
108                r.skipValue();
109            }
110        }
111        r.endObject();
112
113        return builder.build();
114    }
115
116    private static ProfilePhoto parseProfilePhoto(JsonReader r) throws IOException {
117        if (isNextNull(r))
118            return null;
119
120        ProfilePhoto.Builder builder = ProfilePhoto.builder();
121
122        r.beginObject();
123        while (r.hasNext()) {
124            String name = r.nextName();
125            if ("PhotoDescription".equals(name)) {
126                builder.photoDescription(r.nextString());
127            } else if ("PhotoFilename".equals(name)) {
128                builder.photoFilename(r.nextString());
129            } else if ("PhotoName".equals(name)) {
130                builder.photoName(r.nextString());
131            } else if ("PhotoUrl".equals(name)) {
132                builder.photoUrl(r.nextString());
133            } else {
134                r.skipValue();
135            }
136        }
137        r.endObject();
138
139        return builder.build();
140    }
141
142    private static GlobalStats parseGlobalStats(JsonReader r) throws IOException {
143        if (isNextNull(r))
144            return null;
145
146        GlobalStats.Builder builder = GlobalStats.builder();
147
148        r.beginObject();
149        while (r.hasNext()) {
150            String name = r.nextName();
151            if ("AccountsLogged".equals(name)) {
152                builder.accountsLogged(r.nextLong());
153            } else if ("ActiveCaches".equals(name)) {
154                builder.activeCaches(r.nextLong());
155            } else if ("ActiveCountries".equals(name)) {
156                builder.activeCountries(r.nextLong());
157            } else if ("NewLogs".equals(name)) {
158                builder.newLogs(r.nextLong());
159            } else {
160                r.skipValue();
161            }
162        }
163        r.endObject();
164
165        return builder.build();
166    }
167
168    private static TrackableStats parseTrackableStats(JsonReader r) throws IOException {
169        if (isNextNull(r))
170            return null;
171
172        TrackableStats.Builder builder = TrackableStats.builder();
173
174        r.beginObject();
175        while (r.hasNext()) {
176            String name = r.nextName();
177            if ("TrackableFindCount".equals(name)) {
178                builder.trackableFindCount(r.nextInt());
179            } else if ("TrackableFindTypes".equals(name)) {
180                builder.trackableFindTypes(parseList(r));
181            } else if ("TrackableOwnedCount".equals(name)) {
182                builder.trackableOwnedCount(r.nextInt());
183            } else if ("TrackableOwnedTypes".equals(name)) {
184                builder.trackableOwnedTypes(parseList(r));
185            } else {
186                r.skipValue();
187            }
188        }
189        r.endObject();
190
191        return builder.build();
192    }
