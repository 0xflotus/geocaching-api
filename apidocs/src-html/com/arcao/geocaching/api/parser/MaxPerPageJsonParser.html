001package com.arcao.geocaching.api.parser;
002
003import com.arcao.geocaching.api.data.apilimits.MaxPerPage;
004
005import java.io.IOException;
006
007public final class MaxPerPageJsonParser {
008    private MaxPerPageJsonParser() {
009    }
010
011    public static MaxPerPage parse(JsonReader r) throws IOException {
012        MaxPerPage.Builder builder = MaxPerPage.builder();
013
014        r.beginObject();
015        while (r.hasNext()) {
016            String name = r.nextName();
017            if ("Geocaches".equals(name)) {
018                builder.geocaches(r.nextInt());
019            } else if ("GeocacheLogs".equals(name)) {
020                builder.geocacheLogs(r.nextInt());
021            } else if ("Trackables".equals(name)) {
022                builder.trackables(r.nextInt());
023            } else if ("TrackableLogs".equals(name)) {
024                builder.trackableLogs(r.nextInt());
025            } else if ("CacheNotes".equals(name)) {
026                builder.cacheNotes(r.nextInt());
027            } else if ("GalleryImages".equals(name)) {
028                builder.galleryImages(r.nextInt());
029            } else {
030                r.skipValue();
031            }
032        }
033        r.endObject();
034
035        return builder.build();
036    }
037}
