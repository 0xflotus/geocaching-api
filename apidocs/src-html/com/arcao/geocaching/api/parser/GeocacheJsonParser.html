001package com.arcao.geocaching.api.parser;
002
003import com.arcao.geocaching.api.data.Geocache;
004import com.arcao.geocaching.api.data.coordinates.Coordinates;
005import com.google.gson.stream.JsonToken;
006
007import java.io.IOException;
008import java.util.ArrayList;
009import java.util.List;
010
011import static com.arcao.geocaching.api.parser.JsonParserUtil.parseAttributeList;
012import static com.arcao.geocaching.api.parser.JsonParserUtil.parseContainerType;
013import static com.arcao.geocaching.api.parser.JsonParserUtil.parseGeocacheType;
014import static com.arcao.geocaching.api.parser.JsonParserUtil.parseUser;
015
016public final class GeocacheJsonParser {
017    private GeocacheJsonParser() {
018    }
019
020    public static List<Geocache> parseList(JsonReader r) throws IOException {
021        if (r.peek() != JsonToken.BEGIN_ARRAY) {
022            r.skipValue();
023        }
024
025        List<Geocache> list = new ArrayList<Geocache>();
026        r.beginArray();
027
028        while (r.hasNext()) {
029            list.add(parse(r));
030        }
031        r.endArray();
032        return list;
033    }
034
035    private static Geocache parse(JsonReader r) throws IOException {
036        Geocache.Builder builder = Geocache.builder();
037        Coordinates.Builder coordinatesBuilder = Coordinates.builder();
038
039        r.beginObject();
040        while (r.hasNext()) {
041            String name = r.nextName();
042            if ("ID".equals(name)) {
043                builder.id(r.nextLong());
044            } else if ("Code".equals(name)) {
045                builder.code(r.nextString());
046            } else if ("Name".equals(name)) {
047                builder.name(r.nextString());
048            } else if ("Longitude".equals(name)) {
049                coordinatesBuilder.longitude(r.nextDouble());
050            } else if ("Latitude".equals(name)) {
051                coordinatesBuilder.latitude(r.nextDouble());
052            } else if ("CacheType".equals(name)) {
053                builder.geocacheType(parseGeocacheType(r));
054            } else if ("Difficulty".equals(name)) {
055                builder.difficulty((float) r.nextDouble());
056            } else if ("Terrain".equals(name)) {
057                builder.terrain((float) r.nextDouble());
058            } else if ("Owner".equals(name)) {
059                builder.owner(parseUser(r));
060            } else if ("Available".equals(name)) {
061                builder.available(r.nextBoolean());
062            } else if ("Archived".equals(name)) {
063                builder.archived(r.nextBoolean());
064            } else if ("IsPremium".equals(name)) {
065                builder.premium(r.nextBoolean());
066            } else if ("Country".equals(name)) {
067                builder.countryName(r.nextString());
068            } else if ("State".equals(name)) {
069                builder.stateName(r.nextString());
070            } else if ("DateCreated".equals(name)) {
071                builder.createDate(JsonParserUtil.parseJsonDate(r.nextString()));
072            } else if ("PublishDateUtc".equals(name)) {
073                builder.publishDate(JsonParserUtil.parseJsonUTCDate(r.nextString()));
074            } else if ("UTCPlaceDate".equals(name)) {
075                builder.placeDate(JsonParserUtil.parseJsonUTCDate(r.nextString()));
076            } else if ("DateLastUpdate".equals(name)) {
077                builder.lastUpdateDate(JsonParserUtil.parseJsonDate(r.nextString()));
078            } else if ("DateLastVisited".equals(name)) {
079                builder.lastVisitDate(JsonParserUtil.parseJsonDate(r.nextString()));
080            } else if ("PlacedBy".equals(name)) {
081                builder.placedBy(r.nextString());
082            } else if ("ContainerType".equals(name)) {
083                builder.containerType(parseContainerType(r));
084            } else if ("TrackableCount".equals(name)) {
085                builder.trackableCount(r.nextInt());
086            } else if ("HasbeenFoundbyUser".equals(name)) {
087                builder.foundByUser(r.nextBoolean());
088            } else if ("ShortDescription".equals(name)) {
089                builder.shortDescription(r.nextString());
090            } else if ("ShortDescriptionIsHtml".equals(name)) {
091                builder.shortDescriptionHtml(r.nextBoolean());
092            } else if ("LongDescription".equals(name)) {
093                builder.longDescription(r.nextString());
094            } else if ("LongDescriptionIsHtml".equals(name)) {
095                builder.longDescriptionHtml(r.nextBoolean());
096            } else if ("EncodedHints".equals(name)) {
097                builder.hint(r.nextString());
098            } else if ("GeocacheLogs".equals(name)) {
099                builder.geocacheLogs(GeocacheLogJsonParser.parseList(r));
100            } else if ("Trackables".equals(name)) {
101                builder.trackables(TrackableJsonParser.parseList(r));
102            } else if ("AdditionalWaypoints".equals(name)) {
103                builder.waypoints(WaypointJsonParser.parseList(r));
104            } else if ("Attributes".equals(name)) {
105                builder.attributes(parseAttributeList(r));
106            } else if ("UserWaypoints".equals(name)) {
107                builder.userWaypoints(UserWaypointsJsonParser.parseList(r));
108            } else if ("GeocacheNote".equals(name)) {
109                builder.personalNote(r.nextString());
110            } else if ("Images".equals(name)) {
111                builder.images(ImageDataJsonParser.parseList(r));
112            } else if ("FavoritePoints".equals(name)) {
113                builder.favoritePoints(r.nextInt());
114            } else if ("CanCacheBeFavorited".equals(name)) {
115                builder.favoritable(r.nextBoolean());
116            } else if ("FoundDate".equals(name)) {
117                builder.foundDate(JsonParserUtil.parseJsonDate(r.nextString()));
118            } else if ("HasbeenFavoritedbyUser".equals(name)) {
119                builder.favoritedByUser(r.nextBoolean());
120            } else if ("ImageCount".equals(name)) {
121                builder.imageCount(r.nextInt());
122            } else if ("IsRecommended".equals(name)) {
123                builder.recommended(r.nextBoolean());
124            } else if ("TrackableCount".equals(name)) {
125                builder.trackableCount(r.nextInt());
126            } else if ("Url".equals(name)) {
127                builder.url(r.nextString());
128            } else if ("GUID".equals(name)) {
129                builder.guid(r.nextString());
130            } else {
131                r.skipValue();
132            }
133        }
134        r.endObject();
135
136        builder.coordinates(coordinatesBuilder.build());
137
138        return builder.build();
139    }
140}
