001package com.arcao.geocaching.api.parser;
002
003import com.arcao.geocaching.api.data.ImageData;
004import com.google.gson.stream.JsonToken;
005
006import java.io.IOException;
007import java.util.ArrayList;
008import java.util.List;
009
010public final class ImageDataJsonParser {
011    private ImageDataJsonParser() {
012    }
013
014    public static List<ImageData> parseList(JsonReader r) throws IOException {
015        if (r.peek() != JsonToken.BEGIN_ARRAY) {
016            r.skipValue();
017        }
018
019        List<ImageData> list = new ArrayList<ImageData>();
020        r.beginArray();
021
022        while (r.hasNext()) {
023            list.add(parse(r));
024        }
025        r.endArray();
026        return list;
027    }
028
029    private static ImageData parse(JsonReader r) throws IOException {
030        ImageData.Builder builder = ImageData.builder();
031
032        r.beginObject();
033        while (r.hasNext()) {
034            String name = r.nextName();
035            if ("Description".equals(name)) {
036                builder.description(r.nextString());
037            } else if ("MobileUrl".equals(name)) {
038                builder.mobileUrl(r.nextString());
039            } else if ("Name".equals(name)) {
040                builder.name(r.nextString());
041            } else if ("ThumbUrl".equals(name)) {
042                builder.thumbUrl(r.nextString());
043            } else if ("Url".equals(name)) {
044                builder.url(r.nextString());
045            } else {
046                r.skipValue();
047            }
048        }
049        r.endObject();
050
051        return builder.build();
052    }
053}
