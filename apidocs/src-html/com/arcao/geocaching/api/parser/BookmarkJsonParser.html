001package com.arcao.geocaching.api.parser;
002
003import com.arcao.geocaching.api.data.bookmarks.Bookmark;
004import com.google.gson.stream.JsonToken;
005
006import java.io.IOException;
007import java.util.ArrayList;
008import java.util.List;
009
010import static com.arcao.geocaching.api.data.type.GeocacheType.fromId;
011
012public final class BookmarkJsonParser {
013    private BookmarkJsonParser() {
014    }
015
016    public static List<Bookmark> parseList(JsonReader r) throws IOException {
017        if (r.peek() != JsonToken.BEGIN_ARRAY) {
018            r.skipValue();
019        }
020
021        List<Bookmark> list = new ArrayList<Bookmark>();
022        r.beginArray();
023        while (r.hasNext()) {
024            list.add(parse(r));
025        }
026        r.endArray();
027
028        return list;
029    }
030
031    private static Bookmark parse(JsonReader r) throws IOException {
032        Bookmark.Builder builder = Bookmark.builder();
033
034        r.beginObject();
035        while (r.hasNext()) {
036            String fieldName = r.nextName();
037            if ("CacheCode".equals(fieldName)) {
038                builder.cacheCode(r.nextString());
039            } else if ("CacheTitle".equals(fieldName)) {
040                builder.cacheTitle(r.nextString());
041            } else if ("CacheTypeID".equals(fieldName)) {
042                builder.geocacheType(fromId(r.nextInt()));
043            } else {
044                r.skipValue();
045            }
046        }
047        r.endObject();
048
049        return builder.build();
050    }
051}
