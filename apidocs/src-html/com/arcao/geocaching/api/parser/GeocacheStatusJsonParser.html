001package com.arcao.geocaching.api.parser;
002
003import com.arcao.geocaching.api.data.GeocacheStatus;
004import com.arcao.geocaching.api.data.type.GeocacheType;
005import com.google.gson.stream.JsonToken;
006
007import java.io.IOException;
008import java.util.ArrayList;
009import java.util.List;
010
011/**
012 * Created by Krzysztof.Nowacki on 2015.11.13.
013 */
014public final class GeocacheStatusJsonParser {
015    private GeocacheStatusJsonParser() {
016    }
017
018    public static List<GeocacheStatus> parseList(JsonReader r) throws IOException {
019        if (r.peek() != JsonToken.BEGIN_ARRAY) {
020            r.skipValue();
021        }
022
023        List<GeocacheStatus> list = new ArrayList<GeocacheStatus>();
024        r.beginArray();
025        while (r.hasNext()) {
026            list.add(parse(r));
027        }
028        r.endArray();
029        return list;
030    }
031
032    private static GeocacheStatus parse(JsonReader r) throws IOException {
033        GeocacheStatus.Builder builder = GeocacheStatus.builder();
034
035        r.beginObject();
036        while (r.hasNext()) {
037            String name = r.nextName();
038            if ("Archived".equals(name)) {
039                builder.archived(r.nextBoolean());
040            } else if ("Available".equals(name)) {
041                builder.available(r.nextBoolean());
042            } else if ("CacheCode".equals(name)) {
043                builder.cacheCode(r.nextString());
044            } else if ("CacheName".equals(name)) {
045                builder.cacheName(r.nextString());
046            } else if ("CacheType".equals(name)) {
047                builder.cacheType(GeocacheType.fromId(r.nextInt()));
048            } else if ("Premium".equals(name)) {
049                builder.premium(r.nextBoolean());
050            } else if ("TrackableCount".equals(name)) {
051                builder.trackableCount(r.nextInt());
052            } else {
053                r.skipValue();
054            }
055        }
056        r.endObject();
057
058        return builder.build();
059    }
060}
