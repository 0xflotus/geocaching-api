001package com.arcao.geocaching.api.parser;
002
003import com.google.gson.stream.JsonToken;
004
005import java.io.IOException;
006import java.io.Reader;
007
008public class JsonReader {
009    private final com.google.gson.stream.JsonReader r;
010
011    public JsonReader(Reader in) {
012        r = new com.google.gson.stream.JsonReader(in);
013    }
014
015    /**
016     * <p>
017     * Configure this parser to be liberal in what it accepts. By default,
018     * this parser is strict and only accepts JSON as specified by <a
019     * href="http://www.ietf.org/rfc/rfc4627.txt">RFC 4627</a>. Setting the
020     * parser to lenient causes it to ignore the following syntax errors:
021     * </p>
022     * <ul>
023     * <li>Streams that start with the <a href="#nonexecuteprefix">non-execute
024     * prefix</a>, <code>")]}'\n"</code>.
025     * <li>Streams that include multiple top-level values. With strict parsing,
026     * each stream must contain exactly one top-level value.
027     * <li>Top-level values of any type. With strict parsing, the top-level
028     * value must be an object or an array.
029     * <li>Numbers may be {@link Double#isNaN() NaNs} or {@link
030     * Double#isInfinite() infinities}.
031     * <li>End of line comments starting with {@code //} or {@code #} and
032     * ending with a newline character.
033     * <li>C-style comments starting with {@code /*} and ending with
034     * {@code *}{@code /}. Such comments may not be nested.
035     * <li>Names that are unquoted or {@code 'single quoted'}.
036     * <li>Strings that are unquoted or {@code 'single quoted'}.
037     * <li>Array elements separated by {@code ;} instead of {@code ,}.
038     * <li>Unnecessary array separators. These are interpreted as if null
039     * was the omitted value.
040     * <li>Names and values separated by {@code =} or {@code =>} instead of
041     * {@code :}.
042     * <li>Name/value pairs separated by {@code ;} instead of {@code ,}.
043     * </ul>
044     */
045    public void setLenient(boolean lenient) {
046        r.setLenient(lenient);
047    }
048
049    /**
050     * Returns true if this parser is liberal in what it accepts.
051     */
052    public boolean isLenient() {
053        return r.isLenient();
054    }
055
056    /**
057     * Consumes the next token from the JSON stream and asserts that it is the
058     * beginning of a new array.
059     *
060     * @throws IOException When IO error occurs
061     */
062    public void beginArray() throws IOException {
063        r.beginArray();
064    }
065
066    /**
067     * Consumes the next token from the JSON stream and asserts that it is the
068     * end of the current array.
069     *
070     * @throws IOException When IO error occurs
071     */
072    public void endArray() throws IOException {
073        r.endArray();
074    }
075
076    /**
077     * Consumes the next token from the JSON stream and asserts that it is the
078     * beginning of a new object.
079     *
080     * @throws IOException When IO error occurs
081     */
082    public void beginObject() throws IOException {
083        r.beginObject();
084    }
085
086    /**
087     * Consumes the next token from the JSON stream and asserts that it is the
088     * end of the current array.
089     *
090     * @throws IOException When IO error occurs
091     */
092    public void endObject() throws IOException {
093        r.endObject();
094    }
095
096    /**
097     * Returns true if the current array or object has another element.
098     *
099     * @throws IOException When IO error occurs
100     */
101    public boolean hasNext() throws IOException {
102        return r.hasNext();
103    }
104
105    /**
106     * Returns the type of the next token without consuming it.
107     *
108     * @throws IOException When IO error occurs
109     */
110    public JsonToken peek() throws IOException {
111        return r.peek();
112    }
113
114    /**
115     * Returns the next token, a {@link JsonToken#NAME property name}, and
116     * consumes it.
117     *
118     * @throws IOException if the next token in the stream is not a property
119     *                     name.
120     */
121    public String nextName() throws IOException {
122        return r.nextName();
123    }
124
125    /**
126     * Returns the {@link JsonToken#STRING string} value of the next token,
127     * consuming it. If the next token is a number, this method will return its
128     * string form.
129     *
130     * @throws IllegalStateException if the next token is not a string or if
131     *                               this reader is closed.
132     * @throws IOException           when I/O error occurs
133     */
134    public String nextString() throws IOException {
135        return nextString(null);
136    }
137
138    public String nextString(String defaultValue) throws IOException {
139        if (r.peek() == JsonToken.NULL) {
140            r.skipValue();
141            return defaultValue;
142        }
143
144        return r.nextString();
145    }
146
147    /**
148     * Returns the {@link JsonToken#BOOLEAN boolean} value of the next token,
149     * consuming it.
150     *
151     * @throws IllegalStateException if the next token is not a boolean or if
152     *                               this reader is closed.
153     * @throws IOException           when I/O error occurs
154     */
155    public boolean nextBoolean() throws IOException {
156        return nextBoolean(false);
157    }
158
159    public boolean nextBoolean(boolean defaultValue) throws IOException {
160        if (r.peek() == JsonToken.NULL) {
161            r.skipValue();
162            return defaultValue;
163        }
164
165        return r.nextBoolean();
166    }
167
168    /**
169     * Consumes the next token from the JSON stream and asserts that it is a
170     * literal null.
171     *
172     * @throws IllegalStateException if the next token is not null or if this
173     *                               reader is closed.
174     * @throws IOException           when I/O error occurs
175     */
176    public void nextNull() throws IOException {
177        r.nextNull();
178    }
179
180    /**
181     * Returns the {@link JsonToken#NUMBER double} value of the next token,
182     * consuming it. If the next token is a string, this method will attempt to
183     * parse it as a double.
184     *
185     * @throws IllegalStateException if the next token is not a literal value.
186     * @throws NumberFormatException if the next literal value cannot be parsed
187     *                               as a double, or is non-finite.
188     * @throws IOException           when I/O error occurs
189     */
190    public double nextDouble() throws IOException {
191        return nextDouble(0D);
192    }
193
194    public double nextDouble(double defaultValue) throws IOException {
195        if (r.peek() == JsonToken.NULL) {
196            r.skipValue();
197            return defaultValue;
198        }
199
200        return r.nextDouble();
201    }
202
203    /**
204     * Returns the {@link JsonToken#NUMBER long} value of the next token,
205     * consuming it. If the next token is a string, this method will attempt to
206     * parse it as a long. If the next token's numeric value cannot be exactly
207     * represented by a Java {@code long}, this method throws.
208     *
209     * @throws IllegalStateException if the next token is not a literal value.
210     * @throws NumberFormatException if the next literal value cannot be parsed
211     *                               as a number, or exactly represented as a long.
212     * @throws IOException           when I/O error occurs
213     */
214    public long nextLong() throws IOException {
215        return nextLong(0L);
216    }
217
218    public long nextLong(long defaultValue) throws IOException {
219        if (r.peek() == JsonToken.NULL) {
220            r.skipValue();
221            return defaultValue;
222        }
223
224        return r.nextLong();
225    }
226
227    /**
228     * Returns the {@link JsonToken#NUMBER int} value of the next token,
229     * consuming it. If the next token is a string, this method will attempt to
230     * parse it as an int. If the next token's numeric value cannot be exactly
231     * represented by a Java {@code int}, this method throws.
232     *
233     * @throws IllegalStateException if the next token is not a literal value.
234     * @throws NumberFormatException if the next literal value cannot be parsed
235     *                               as a number, or exactly represented as an int.
236     * @throws IOException           when I/O error occurs
237     */
238    public int nextInt() throws IOException {
239        return nextInt(0);
240    }
241
242    public int nextInt(int defaultValue) throws IOException {
243        if (r.peek() == JsonToken.NULL) {
244            r.skipValue();
245            return defaultValue;
246        }
247
248        return r.nextInt();
249    }
250
251    /**
252     * Closes this JSON reader and the underlying {@link Reader}.
253     *
254     * @throws IOException when I/O error occurs
255     */
256    public void close() throws IOException {
257        r.close();
258    }
259
260    /**
261     * Skips the next value recursively. If it is an object or array, all nested
262     * elements are skipped. This method is intended for use when the JSON token
263     * stream contains unrecognized or unhandled values.
264     *
265     * @throws IOException when I/O error occurs
266     */
267    public void skipValue() throws IOException {
268        r.skipValue();
269    }
270
271    @Override
272    public String toString() {
273        return r.toString();
274    }
275}
