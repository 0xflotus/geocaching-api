001package com.arcao.geocaching.api.parser;
002
003import com.arcao.geocaching.api.data.TrackableLog;
004import com.arcao.geocaching.api.data.coordinates.Coordinates;
005import com.google.gson.stream.JsonToken;
006
007import java.io.IOException;
008import java.util.ArrayList;
009import java.util.List;
010
011import static com.arcao.geocaching.api.parser.JsonParserUtil.parseJsonDate;
012import static com.arcao.geocaching.api.parser.JsonParserUtil.parseJsonUTCDate;
013import static com.arcao.geocaching.api.parser.JsonParserUtil.parseTrackableLogType;
014import static com.arcao.geocaching.api.parser.JsonParserUtil.parseUser;
015
016public final class TrackableLogJsonParser {
017    private TrackableLogJsonParser() {
018    }
019
020    public static List<TrackableLog> parseList(JsonReader r) throws IOException {
021        if (r.peek() != JsonToken.BEGIN_ARRAY) {
022            r.skipValue();
023        }
024
025        List<TrackableLog> list = new ArrayList<TrackableLog>();
026        r.beginArray();
027        while (r.hasNext()) {
028            list.add(parse(r));
029        }
030        r.endArray();
031
032        return list;
033    }
034
035    public static TrackableLog parse(JsonReader r) throws IOException {
036        TrackableLog.Builder builder = TrackableLog.builder();
037        Coordinates.Builder updatedCoordinatesBuilder = Coordinates.builder();
038
039        r.beginObject();
040        while (r.hasNext()) {
041            String name = r.nextName();
042            if ("CacheID".equals(name)) {
043                builder.cacheId(r.nextInt());
044            } else if ("Code".equals(name)) {
045                builder.code(r.nextString());
046            } else if ("ID".equals(name)) {
047                builder.id(r.nextInt());
048            } else if ("Images".equals(name)) {
049                builder.images(ImageDataJsonParser.parseList(r));
050            } else if ("IsArchived".equals(name)) {
051                builder.archived(r.nextBoolean());
052            } else if ("LogGuid".equals(name)) {
053                builder.guid(r.nextString());
054            } else if ("LogText".equals(name)) {
055                builder.text(r.nextString());
056            } else if ("LogType".equals(name)) {
057                builder.type(parseTrackableLogType(r));
058            } else if ("LoggedBy".equals(name)) {
059                builder.loggedBy(parseUser(r));
060            } else if ("UTCCreateDate".equals(name)) {
061                builder.created(parseJsonUTCDate(r.nextString()));
062            } else if ("UpdatedLatitude".equals(name)) {
063                updatedCoordinatesBuilder.latitude(r.nextDouble());
064            } else if ("UpdatedLongitude".equals(name)) {
065                updatedCoordinatesBuilder.longitude(r.nextDouble());
066            } else if ("Url".equals(name)) {
067                builder.url(r.nextString());
068            } else if ("VisitDate".equals(name)) {
069                builder.visited(parseJsonDate(r.nextString()));
070            } else {
071                r.skipValue();
072            }
073        }
074        r.endObject();
075
076        builder.updatedCoordinates(updatedCoordinatesBuilder.build());
077
078        return builder.build();
079    }
080}
