001package com.arcao.geocaching.api.filter;
002
003import com.google.gson.stream.JsonWriter;
004
005import org.jetbrains.annotations.NotNull;
006import org.jetbrains.annotations.Nullable;
007
008import java.io.IOException;
009
010public class GeocacheExclusionsFilter implements Filter {
011    private static final String NAME = "GeocacheExclusions";
012
013    @Nullable private final Boolean archived;
014    @Nullable private final Boolean available;
015    @Nullable private final Boolean hasCorrectedCoordinates;
016    @Nullable private final Boolean hasPersonalCacheNote;
017    @Nullable private final Boolean premium;
018    @Nullable private final Boolean published;
019
020    public GeocacheExclusionsFilter(@Nullable Boolean archived, @Nullable Boolean available, @Nullable Boolean hasCorrectedCoordinates, @Nullable Boolean hasPersonalCacheNote, @Nullable Boolean premium, @Nullable Boolean published) {
021        this.archived = archived;
022        this.available = available;
023        this.hasCorrectedCoordinates = hasCorrectedCoordinates;
024        this.hasPersonalCacheNote = hasPersonalCacheNote;
025        this.premium = premium;
026        this.published = published;
027    }
028
029    @NotNull
030    @Override
031    public String name() {
032        return NAME;
033    }
034
035    @Override
036    public boolean valid() {
037        return archived != null || available != null || hasCorrectedCoordinates != null || hasPersonalCacheNote != null || premium != null || published != null;
038    }
039
040    @Override
041    public void writeJson(@NotNull JsonWriter w) throws IOException {
042        w.name(NAME);
043        w.beginObject();
044        if (archived != null)
045            w.name("Archived").value(archived);
046        if (available != null)
047            w.name("Available").value(available);
048        if (hasCorrectedCoordinates != null)
049            w.name("HasCorrectedCoordinates").value(hasCorrectedCoordinates);
050        if (hasPersonalCacheNote != null)
051            w.name("HasPersonalCacheNote").value(hasPersonalCacheNote);
052        if (premium != null)
053            w.name("Premium").value(premium);
054        if (published != null)
055            w.name("Published").value(published);
056        w.endObject();
057    }
058}
