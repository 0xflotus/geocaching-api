001package com.arcao.geocaching.api.filter;
002
003import com.google.gson.stream.JsonWriter;
004
005import org.jetbrains.annotations.NotNull;
006
007import java.io.IOException;
008
009public class NotFoundByUsersFilter implements Filter {
010    private static final String NAME = "NotFoundByUsers";
011
012    @NotNull private final String[] userNames;
013
014    public NotFoundByUsersFilter(@NotNull String... userNames) {
015        this.userNames = userNames;
016    }
017
018    @NotNull
019    @Override
020    public String name() {
021        return NAME;
022    }
023
024    @Override
025    public boolean valid() {
026        if (userNames.length == 0)
027            return false;
028
029        for (String userName : userNames) {
030            if (userName != null && !userName.isEmpty())
031                return true;
032        }
033
034        return false;
035    }
036
037    @Override
038    public void writeJson(@NotNull JsonWriter w) throws IOException {
039        w.name(NAME);
040        w.beginObject();
041        w.name("UserNames");
042        w.beginArray();
043        for (String userName : userNames) {
044            if (userName != null && !userName.isEmpty())
045                w.value(userName);
046        }
047        w.endArray();
048        w.endObject();
049    }
050}
