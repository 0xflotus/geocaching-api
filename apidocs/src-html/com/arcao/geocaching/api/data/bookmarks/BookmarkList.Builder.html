001package com.arcao.geocaching.api.data.bookmarks;
002
003import com.google.auto.value.AutoValue;
004
005import java.io.Serializable;
006
007@AutoValue
008public abstract class BookmarkList implements Serializable {
009    private static final long serialVersionUID = 2322622811124797813L;
010
011    public abstract int id();
012
013    public abstract String guid();
014
015    public abstract String name();
016
017    public abstract String description();
018
019    public abstract int itemCount();
020
021    public abstract boolean shared();
022
023    public abstract boolean publicList();
024
025    public abstract boolean archived();
026
027    public abstract boolean special();
028
029    public abstract int type();
030
031    public static Builder builder() {
032        return new AutoValue_BookmarkList.Builder();
033    }
034
035    @AutoValue.Builder
036    public abstract static class Builder {
037        public abstract Builder id(int id);
038
039        public abstract Builder guid(String guid);
040
041        public abstract Builder name(String name);
042
043        public abstract Builder description(String description);
044
045        public abstract Builder itemCount(int itemCount);
046
047        public abstract Builder shared(boolean shared);
048
049        public abstract Builder archived(boolean archived);
050
051        public abstract Builder special(boolean special);
052
053        public abstract Builder type(int type);
054
055        public abstract Builder publicList(boolean publicList);
056
057        public abstract BookmarkList build();
058    }
059}
