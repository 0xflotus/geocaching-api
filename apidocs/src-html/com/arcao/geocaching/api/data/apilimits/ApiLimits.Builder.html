001package com.arcao.geocaching.api.data.apilimits;
002
003import com.arcao.geocaching.api.data.type.MemberType;
004import com.google.auto.value.AutoValue;
005
006import java.io.Serializable;
007import java.util.List;
008
009/**
010 * Container class for API limits related to currently logged user
011 *
012 * @author arcao
013 */
014@AutoValue
015public abstract class ApiLimits implements Serializable {
016    private static final long serialVersionUID = -3608995080972521881L;
017
018    /**
019     * Returns limits for downloading caches
020     *
021     * @return cache limit object
022     */
023    public abstract List<CacheLimit> cacheLimits();
024
025    /**
026     * Returns if the cache limits are enforced
027     *
028     * @return are enforced
029     */
030    public abstract boolean enforceCacheLimits();
031
032    /**
033     * Returns if the cache limits are enforced for a lite caches
034     *
035     * @return are enforced
036     */
037    public abstract boolean enforceLiteCacheLimits();
038
039    /**
040     * Returns if the method limits are enforced
041     *
042     * @return are enforced
043     */
044    public abstract boolean enforceMethodLimits();
045
046    /**
047     * Returns a membership type for which are limits applied
048     *
049     * @return member type
050     */
051    public abstract MemberType forMembershipType();
052
053    /**
054     * Returns a license key
055     *
056     * @return license key
057     */
058    public abstract String licenseKey();
059
060    /**
061     * Returns a cache limits for lite caches
062     *
063     * @return cache limits
064     */
065    public abstract List<CacheLimit> liteCacheLimits();
066
067    /**
068     * Returns the number of max calls to be performed with same IP per minute
069     *
070     * @return number of max calls
071     */
072    public abstract long maxCallsbyIPIn1Minute();
073
074    /**
075     * Returns the method limits
076     *
077     * @return method limits
078     */
079    public abstract List<MethodLimit> methodLimits();
080
081    /**
082     * Returns whereas limits are restricted by IP
083     *
084     * @return are restricted by IP
085     */
086    public abstract boolean restrictByIp();
087
088    /**
089     * Returns if counts of IP are validated
090     *
091     * @return count of IP validated
092     */
093    public abstract boolean validateIpCounts();
094
095    public static Builder builder() {
096        return new AutoValue_ApiLimits.Builder();
097    }
098
099    @AutoValue.Builder
100    public abstract static class Builder {
101        public abstract Builder cacheLimits(List<CacheLimit> cacheLimits);
102        public abstract Builder enforceCacheLimits(boolean enforceCacheLimits);
103        public abstract Builder enforceLiteCacheLimits(boolean enforceLiteCacheLimits);
104        public abstract Builder enforceMethodLimits(boolean enforceMethodLimits);
105        public abstract Builder forMembershipType(MemberType forMembershipType);
106        public abstract Builder licenseKey(String licenseKey);
107        public abstract Builder liteCacheLimits(List<CacheLimit> liteCacheLimits);
108        public abstract Builder maxCallsbyIPIn1Minute(long maxCallsbyIPIn1Minute);
109        public abstract Builder methodLimits(List<MethodLimit> methodLimits);
110        public abstract Builder restrictByIp(boolean restrictByIp);
111        public abstract Builder validateIpCounts(boolean validateIpCounts);
112        public abstract ApiLimits build();
113    }
114}
