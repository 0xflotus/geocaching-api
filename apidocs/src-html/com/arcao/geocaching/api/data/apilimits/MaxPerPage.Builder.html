001package com.arcao.geocaching.api.data.apilimits;
002
003import com.google.auto.value.AutoValue;
004
005import java.io.Serializable;
006
007/**
008 * Container class for a max per page limit information
009 *
010 * @author arcao
011 */
012@AutoValue
013public abstract class MaxPerPage implements Serializable {
014    private static final long serialVersionUID = -5594435197387844111L;
015
016    /**
017     * Returns a max per page value for retrieving Geocaches
018     *
019     * @return max per page value for Geocaches
020     */
021    public abstract int geocaches();
022
023    /**
024     * Returns a max per page value for retrieving Geocache logs
025     *
026     * @return max per page value for Geocache logs
027     */
028    public abstract int geocacheLogs();
029
030    /**
031     * Returns a max per page value for retrieving Trackables
032     *
033     * @return max per page value for Trackables
034     */
035    public abstract int trackables();
036
037    /**
038     * Returns a max per page value for retrieving Trackable logs
039     *
040     * @return max per page value for Trackable logs
041     */
042    public abstract int trackableLogs();
043
044    /**
045     * Returns a max per page value for retrieving Cache notes
046     *
047     * @return max per page value for Cache notes
048     */
049    public abstract int cacheNotes();
050
051    /**
052     * Returns a max per page value for retrieving Gallery images
053     *
054     * @return max per page value for Gallery images
055     */
056    public abstract int galleryImages();
057
058    public static Builder builder() {
059        return new AutoValue_MaxPerPage.Builder();
060    }
061
062    @AutoValue.Builder
063    public abstract static class Builder {
064        public abstract Builder geocaches(int geocaches);
065
066        public abstract Builder geocacheLogs(int geocacheLogs);
067
068        public abstract Builder trackables(int trackables);
069
070        public abstract Builder trackableLogs(int trackableLogs);
071
072        public abstract Builder cacheNotes(int cacheNotes);
073
074        public abstract Builder galleryImages(int galleryImages);
075
076        public abstract MaxPerPage build();
077    }
078}
