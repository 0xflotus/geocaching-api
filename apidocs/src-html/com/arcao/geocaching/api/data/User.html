001package com.arcao.geocaching.api.data;
002
003
004import com.arcao.geocaching.api.data.coordinates.Coordinates;
005import com.arcao.geocaching.api.data.type.MemberType;
006import com.google.auto.value.AutoValue;
007
008import org.jetbrains.annotations.Nullable;
009
010import java.io.Serializable;
011
012@AutoValue
013public abstract class User implements Serializable {
014    private static final long serialVersionUID = 1808891631464643511L;
015
016    public abstract long id();
017
018    public abstract String publicGuid();
019
020    public abstract String userName();
021
022    public abstract String avatarUrl();
023
024    @Nullable public abstract Coordinates homeCoordinates();
025
026    public abstract boolean admin();
027
028    @Nullable public abstract MemberType memberType();
029
030    public abstract int findCount();
031
032    public abstract int hideCount();
033
034    public abstract int galleryImageCount();
035
036    public static Builder builder() {
037        return new AutoValue_User.Builder().galleryImageCount(0);
038    }
039
040    @AutoValue.Builder
041    public abstract static class Builder {
042        public abstract Builder id(long id);
043
044        public abstract Builder publicGuid(String publicGuid);
045
046        public abstract Builder userName(String userName);
047
048        public abstract Builder avatarUrl(String avatarUrl);
049
050        public abstract Builder homeCoordinates(Coordinates homeCoordinates);
051
052        public abstract Builder admin(boolean admin);
053
054        public abstract Builder memberType(MemberType memberType);
055
056        public abstract Builder findCount(int findCount);
057
058        public abstract Builder hideCount(int hideCount);
059
060        public abstract Builder galleryImageCount(int galleryImageCount);
061
062        public abstract User build();
063    }
064}
