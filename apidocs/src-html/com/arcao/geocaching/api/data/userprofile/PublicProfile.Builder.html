001package com.arcao.geocaching.api.data.userprofile;
002
003import com.google.auto.value.AutoValue;
004
005import java.io.Serializable;
006import java.util.Date;
007
008@AutoValue
009public abstract class PublicProfile implements Serializable {
010    private static final long serialVersionUID = 7624712344301138677L;
011
012    public abstract String forumTitle();
013
014    public abstract Date lastVisit();
015
016    public abstract String location();
017
018    public abstract Date memberSince();
019
020    public abstract String occupation();
021
022    public abstract ProfilePhoto profilePhoto();
023
024    public abstract String profileText();
025
026    public static Builder builder() {
027        return new AutoValue_PublicProfile.Builder();
028    }
029
030    @AutoValue.Builder
031    public abstract static class Builder {
032        public abstract Builder forumTitle(String forumTitle);
033
034        public abstract Builder lastVisit(Date lastVisit);
035
036        public abstract Builder location(String location);
037
038        public abstract Builder memberSince(Date memberSince);
039
040        public abstract Builder occupation(String occupation);
041
042        public abstract Builder profilePhoto(ProfilePhoto profilePhoto);
043
044        public abstract Builder profileText(String profileText);
045
046        public abstract PublicProfile build();
047    }
048}
