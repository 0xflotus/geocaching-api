001package com.arcao.geocaching.api.data.userprofile;
002
003import com.arcao.geocaching.api.data.Trackable;
004import com.google.auto.value.AutoValue;
005
006import java.io.Serializable;
007import java.util.List;
008
009@AutoValue
010public abstract class TrackableStats implements Serializable {
011    private static final long serialVersionUID = 8539963736459413400L;
012
013    public abstract int trackableFindCount();
014    public abstract List<Trackable> trackableFindTypes();
015    public abstract int trackableOwnedCount();
016    public abstract List<Trackable> trackableOwnedTypes();
017
018    public static Builder builder() {
019        return new AutoValue_TrackableStats.Builder();
020    }
021
022    @AutoValue.Builder
023    public abstract static class Builder {
024        public abstract Builder trackableFindCount(int trackableFindCount);
025
026        public abstract Builder trackableFindTypes(List<Trackable> trackableFindTypes);
027
028        public abstract Builder trackableOwnedCount(int trackableOwnedCount);
029
030        public abstract Builder trackableOwnedTypes(List<Trackable> trackableOwnedTypes);
031
032        public abstract TrackableStats build();
033    }
034}
