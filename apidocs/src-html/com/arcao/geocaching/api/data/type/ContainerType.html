001package com.arcao.geocaching.api.data.type;
002
003import org.jetbrains.annotations.NotNull;
004import org.jetbrains.annotations.Nullable;
005
006public enum ContainerType {
007    NotChosen("Not chosen", 1),
008    Micro("Micro", 2),
009    Small("Small", 8),
010    Regular("Regular", 3),
011    Large("Large", 4),
012    Huge("Huge", 5),
013    Other("Other", 6);
014
015    /**
016     * Friendly name
017     */
018    @NotNull public final String name;
019    /**
020     * Groundspeak Id
021     */
022    public final int id;
023
024    ContainerType(@NotNull String name, int id) {
025        this.name = name;
026        this.id = id;
027    }
028
029    @Override
030    public String toString() {
031        return name;
032    }
033
034    /**
035     * Get an ContainerType from Friendly name. If Friendly name is not valid, the Other value is returned.
036     *
037     * @param name Friendly name
038     * @return Container Type
039     */
040    @NotNull
041    public static ContainerType fromName(@Nullable String name) {
042        for (ContainerType type : values()) {
043            if (type.name.equals(name))
044                return type;
045        }
046
047        return Other;
048    }
049
050    /**
051     * Get an ContainerType from Groundspeak Id. If Groundspeak Id is not valid, the Other value is returned.
052     *
053     * @param id Groundspeak Id
054     * @return Container Type
055     */
056    @NotNull
057    public static ContainerType fromId(int id) {
058        for (ContainerType type : values()) {
059            if (type.id == id)
060                return type;
061        }
062
063        return Other;
064    }
065}
