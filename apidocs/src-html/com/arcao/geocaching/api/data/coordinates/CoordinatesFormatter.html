015/**
016 * A class for formating geographic coordinates represented by Coordinates class
017 * to user readable format.
018 *
019 * @author arcao
020 * @since 1.5
021 */
022public class CoordinatesFormatter {
023    static final Logger logger = LoggerFactory.getLogger(CoordinatesFormatter.class);
024
025    /**
026     * Example: "10,123456 -0,123456"
027     */
028    public static final int LAT_LON_DECDEGREE = 0;
029
030    /**
031     * Example: "10.123456,-0.123456" (unlocalized)
032     */
033    public static final int LAT_LON_DECDEGREE_COMMA = 1;
034
035    /**
036     * Example: "N 10° 12,345 · W 5° 12,345"
037     */
038    public static final int LAT_LON_DECMINUTE = 2;
039
040    /**
041     * Example: "N 10° 12.345 W 5° 12.345" (unlocalized)
042     */
043    public static final int LAT_LON_DECMINUTE_RAW = 3;
044
045    /**
046     * Example: "N 10° 12' 34" W 5° 12' 34""
047     */
048    public static final int LAT_LON_DECSECOND = 4;
049
050    /**
051     * Example: "-0.123456" (unlocalized latitude)
052     */
053    public static final int LAT_DECDEGREE_RAW = 5;
054
055    /**
056     * Example: "N 10° 12,345"
057     */
058    public static final int LAT_DECMINUTE = 6;
059
060    /**
061     * Example: "N 10 12,345"
062     */
063    public static final int LAT_DECMINUTE_RAW = 7;
064
065    /**
066     * Example: "-0.123456" (unlocalized longitude)
067     */
068    public static final int LON_DECDEGREE_RAW = 8;
069
070    /**
071     * Example: "W 5° 12,345"
072     */
073    public static final int LON_DECMINUTE = 9;
074
075    /**
076     * Example: "W 5 12,345"
077     */
078    public static final int LON_DECMINUTE_RAW = 10;
079
080    private final int format;
081    private final Locale locale;
082
083    /**
084     * Create a new instance of formatter with predefined format and system
085     * default locale for formating numbers.
086     *
087     * @param format format to use
088     */
089    public CoordinatesFormatter(int format) {
090        this(format, Locale.getDefault());
091    }
092
093    /**
094     * Create a new instance of formatter with predefined format and locale for
095     * formating numbers.
096     *
097     * @param format format to use
098     * @param locale locale to use for formating numbers
099     */
100    public CoordinatesFormatter(int format, Locale locale) {
101        this.format = format;
102        this.locale = locale;
103    }
104
105    /**
106     * Formats the Coordinates.
107     *
108     * @param coords the Coordinates to format
109     * @return the formatted coordinates
110     */
111    public String format(final Coordinates coords) {
112        final double latSigned = coords.latitude();
113        final double lonSigned = coords.longitude();
114
115        DMM dmm = new DMM(latSigned, lonSigned);
116        DMS dms = new DMS(latSigned, lonSigned);
117
118        switch (format) {
119            case LAT_LON_DECDEGREE:
120                return String.format(locale, "%.6f %.6f", latSigned, lonSigned);
121
122            case LAT_LON_DECDEGREE_COMMA:
123                return String.format((Locale) null, "%.6f,%.6f", latSigned, lonSigned);
124
125            case LAT_LON_DECMINUTE:
126                return String.format(locale, "%c %d° %06.3f · %c %d° %06.3f",
127                        dmm.latDir, dmm.latDeg, dmm.latMinRaw, dmm.lonDir, dmm.lonDeg, dmm.lonMinRaw);
128
129            case LAT_LON_DECMINUTE_RAW:
130                return String.format((Locale) null, "%c %d° %06.3f %c %d° %06.3f",
131                        dmm.latDir, dmm.latDeg, dmm.latMinRaw, dmm.lonDir, dmm.lonDeg, dmm.lonMinRaw);
132
133            case LAT_LON_DECSECOND:
134                return String.format(locale, "%c %d° %d' %06.3f\" · %c %d° %d' %06.3f\"",
135                        dms.latDir, dms.latDeg, dms.latMin, dms.latSecRaw, dms.lonDir, dms.lonDeg, dms.lonMin, dms.lonSecRaw);
136
137            case LAT_DECDEGREE_RAW:
138                return String.format((Locale) null, "%.6f", latSigned);
139
140            case LAT_DECMINUTE:
141                return String.format(locale, "%c %d° %06.3f", dmm.latDir, dmm.latDeg, dmm.latMinRaw);
142
143            case LAT_DECMINUTE_RAW:
144                return String.format(locale, "%c %d %06.3f", dmm.latDir, dmm.latDeg, dmm.latMinRaw);
145
146            case LON_DECDEGREE_RAW:
147                return String.format((Locale) null, "%.6f", lonSigned);
148
149            case LON_DECMINUTE:
150                return String.format(locale, "%c %d° %06.3f", dmm.lonDir, dmm.lonDeg, dmm.lonMinRaw);
151
152            case LON_DECMINUTE_RAW:
153                return String.format(locale, "%c %d %06.3f", dmm.lonDir, dmm.lonDeg, dmm.lonMinRaw);
154        }
155
156        // Keep the compiler happy even though it cannot happen
157        return null;
158    }

160    /* Constant values needed for calculation */
161    private static final double D60 = 60.0d;
162    private static final double D1000 = 1000.0d;
163    private static final double D3600 = 3600.0d;
164    static final BigDecimal BD_SIXTY = BigDecimal.valueOf(D60);
165    static final BigDecimal BD_THOUSAND = BigDecimal.valueOf(D1000);
166    static final BigDecimal BD_ONEHOUNDREDTHOUSAND = BigDecimal.valueOf(100000.0d);
167
168    /**
169     * Value type for the direction.
170     */
171    protected static class Direction {
172        /**
173         * latitude direction, 'N' or 'S'
174         */
175        public final char latDir;
176        /**
177         * longitude direction, 'E' or 'W'
178         */
179        public final char lonDir;
180
181        protected Direction(final double latSigned, final double lonSigned) {
182            latDir = latSigned < 0 ? 'S' : 'N';
183            lonDir = lonSigned < 0 ? 'W' : 'E';
184        }
185
186        protected static String addZeros(final int value, final int len) {
187            return StringUtils.leftPad(Integer.toString(value), len, '0');
188        }
189    }
190
191    /**
192     * Helper class to calculate degrees for coordinates
193     */
194    protected static final class DDD extends Direction {
195
196        /**
197         * latitude degree value
198         */
199        public final int latDeg;
200        /**
201         * fractional part of the latitude degree value
202         */
203        public final int latDegFrac;
204
205        public final int lonDeg;
206        public final int lonDegFrac;
207
208        public DDD(final double latSigned, final double lonSigned) {
209            super(latSigned, lonSigned);
210            BigDecimal bdLat = BigDecimal.valueOf(latSigned).abs();
211            latDeg = bdLat.intValue();
212            BigDecimal bdLatFrac = bdLat.subtract(BigDecimal.valueOf(latDeg)).multiply(BD_ONEHOUNDREDTHOUSAND);
213            latDegFrac = bdLatFrac.setScale(0, RoundingMode.HALF_UP).intValue();
214
215            BigDecimal bdlon = BigDecimal.valueOf(lonSigned).abs();
216            lonDeg = bdlon.intValue();
217            BigDecimal bdLonFrac = bdlon.subtract(BigDecimal.valueOf(lonDeg)).multiply(BD_ONEHOUNDREDTHOUSAND);
218            lonDegFrac = bdLonFrac.setScale(0, RoundingMode.HALF_UP).intValue();
219        }
220
221        public static Coordinates createCoordinates(final String latDir, final String latDeg, final String latDegFrac,
222                                                    final String lonDir, final String lonDeg, final String lonDegFrac) {
223            double lat = 0.0d;
224            double lon = 0.0d;
225            try {
226                lat = Double.parseDouble(latDeg + "." + addZeros(Integer.parseInt(latDegFrac), 5));
227                lon = Double.parseDouble(lonDeg + "." + addZeros(Integer.parseInt(lonDegFrac), 5));
228            } catch (NumberFormatException e) {
229                logger.error(e.getMessage(), e);
230            }
231            lat *= "S".equalsIgnoreCase(latDir) ? -1 : 1;
232            lon *= "W".equalsIgnoreCase(lonDir) ? -1 : 1;
233            return Coordinates.create(lat, lon);
234        }
235    }

237    /**
238     * Helper class to calculate degree and minutes for coordinates
239     */
240    protected static final class DMM extends Direction {
241
242        public final int latDeg;
243        public final double latMinRaw;
244        public final int latMin;
245        public final int latMinFrac;
246
247        public final int lonDeg;
248        public final double lonMinRaw;
249        public final int lonMin;
250        public final int lonMinFrac;
251
252        public DMM(final double latSigned, final double lonSigned) {
253            super(latSigned, lonSigned);
254            BigDecimal bdLat = BigDecimal.valueOf(latSigned).abs();
255            latDeg = bdLat.intValue();
256            BigDecimal bdLatMin = bdLat.subtract(BigDecimal.valueOf(latDeg)).multiply(BD_SIXTY);
257            // Rounding here ...
258            bdLatMin = bdLatMin.setScale(3, RoundingMode.HALF_UP);
259            latMinRaw = bdLatMin.doubleValue();
260            latMin = bdLatMin.intValue();
261            BigDecimal bdLatMinFrac = bdLatMin.subtract(BigDecimal.valueOf(latMin)).multiply(BD_THOUSAND);
262            latMinFrac = bdLatMinFrac.setScale(0, RoundingMode.HALF_UP).intValue();
263
264            BigDecimal bdlon = BigDecimal.valueOf(lonSigned).abs();
265            lonDeg = bdlon.intValue();
266            BigDecimal bdLonMin = bdlon.subtract(BigDecimal.valueOf(lonDeg)).multiply(BD_SIXTY);
267            // Rounding here ...
268            bdLonMin = bdLonMin.setScale(3, RoundingMode.HALF_UP);
269            lonMinRaw = bdLonMin.doubleValue();
270            lonMin = bdLonMin.intValue();
271            BigDecimal bdLonMinFrac = bdLonMin.subtract(BigDecimal.valueOf(lonMin)).multiply(BD_THOUSAND);
272            lonMinFrac = bdLonMinFrac.setScale(0, RoundingMode.HALF_UP).intValue();
273        }
274
275        public static Coordinates createCoordinates(final String latDir, final String latDeg, final String latMin, final String latMinFrac,
276                                                    final String lonDir, final String lonDeg, final String lonMin, final String lonMinFrac) {
277            double lat = 0.0d;
278            double lon = 0.0d;
279            try {
280                lat = Double.parseDouble(latDeg) + Double.parseDouble(latMin + "." + addZeros(Integer.parseInt(latMinFrac), 3)) / D60;
281                lon = Double.parseDouble(lonDeg) + Double.parseDouble(lonMin + "." + addZeros(Integer.parseInt(lonMinFrac), 3)) / D60;
282            } catch (NumberFormatException e) {
283                logger.error(e.getMessage(), e);
284            }
285            lat *= "S".equalsIgnoreCase(latDir) ? -1 : 1;
286            lon *= "W".equalsIgnoreCase(lonDir) ? -1 : 1;
287            return Coordinates.create(lat, lon);
288        }
289    }

291    /**
292     * Helper class to calculate degree, minutes and seconds for coordinates
293     */
294    protected static final class DMS extends Direction {
295
296        public final int latDeg;
297        public final int latMin;
298        public final double latSecRaw;
299        public final int latSec;
300        public final int latSecFrac;
301
302        public final int lonDeg;
303        public final int lonMin;
304        public final double lonSecRaw;
305        public final int lonSec;
306        public final int lonSecFrac;
307
308        public DMS(final double latSigned, final double lonSigned) {
309            super(latSigned, lonSigned);
310            BigDecimal bdLat = BigDecimal.valueOf(latSigned).abs();
311            latDeg = bdLat.intValue();
312            BigDecimal bdLatMin = bdLat.subtract(BigDecimal.valueOf(latDeg)).multiply(BD_SIXTY);
313            latMin = bdLatMin.intValue();
314            BigDecimal bdLatSec = bdLatMin.subtract(BigDecimal.valueOf(latMin)).multiply(BD_SIXTY);
315            // Rounding here ...
316            bdLatSec = bdLatSec.setScale(3, RoundingMode.HALF_UP);
317            latSecRaw = bdLatSec.doubleValue();
318            latSec = bdLatSec.intValue();
319            BigDecimal bdLatSecFrac = bdLatSec.subtract(BigDecimal.valueOf(latSec)).multiply(BD_THOUSAND);
320            latSecFrac = bdLatSecFrac.setScale(0, RoundingMode.HALF_UP).intValue();
321
322            BigDecimal bdlon = BigDecimal.valueOf(lonSigned).abs();
323            lonDeg = bdlon.intValue();
324            BigDecimal bdLonMin = bdlon.subtract(BigDecimal.valueOf(lonDeg)).multiply(BD_SIXTY);
325            lonMin = bdLonMin.intValue();
326            BigDecimal bdLonSec = bdLonMin.subtract(BigDecimal.valueOf(lonMin)).multiply(BD_SIXTY);
327            // Rounding here ...
328            bdLonSec = bdLonSec.setScale(3, RoundingMode.HALF_UP);
329            lonSecRaw = bdLonSec.doubleValue();
330            lonSec = bdLonSec.intValue();
331            BigDecimal bdLonSecFrac = bdLonSec.subtract(BigDecimal.valueOf(lonSec)).multiply(BD_THOUSAND);
332            lonSecFrac = bdLonSecFrac.setScale(0, RoundingMode.HALF_UP).intValue();
333        }
334
335        public static Coordinates createCoordinates(final String latDir, final String latDeg, final String latMin, final String latSec, final String latSecFrac,
336                                                    final String lonDir, final String lonDeg, final String lonMin, final String lonSec, final String lonSecFrac) {
337            double lat = 0.0d;
338            double lon = 0.0d;
339            try {
340                lat = Double.parseDouble(latDeg) + Double.parseDouble(latMin) / D60 + Double.parseDouble(latSec + "." + addZeros(Integer.parseInt(latSecFrac), 3))
341                        / D3600;
342                lon = Double.parseDouble(lonDeg) + Double.parseDouble(lonMin) / D60 + Double.parseDouble(lonSec + "." + addZeros(Integer.parseInt(lonSecFrac), 3))
343                        / D3600;
344            } catch (NumberFormatException e) {
345                logger.error(e.getMessage(), e);
346            }
347            lat *= "S".equalsIgnoreCase(latDir) ? -1 : 1;
348            lon *= "W".equalsIgnoreCase(lonDir) ? -1 : 1;
349            return Coordinates.create(lat, lon);
350        }
