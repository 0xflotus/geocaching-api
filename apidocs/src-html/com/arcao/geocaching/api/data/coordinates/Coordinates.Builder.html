001package com.arcao.geocaching.api.data.coordinates;

003import com.google.auto.value.AutoValue;
004
005import org.jetbrains.annotations.Nullable;
006
007import java.io.Serializable;
008import java.text.ParseException;
009
010/**
011 * A class representing a geographic WGS-84 coordinates. Coordinates consist of
012 * latitude and longitude.
013 *
014 * @author arcao
015 * @since 1.5
016 */
017@SuppressWarnings("EqualsAndHashcode")
018@AutoValue
019public abstract class Coordinates implements Serializable {
020    private static final long serialVersionUID = 1044000671539652241L;
021    private static final double PRECISION = 1e-8;
022
023    private static final double AVERAGE_RADIUS_OF_EARTH = 6372797.560856;
024    private static final CoordinatesFormatter LAT_LON_DECMINUTE_FORMAT = new CoordinatesFormatter(CoordinatesFormatter.LAT_LON_DECMINUTE);
025
026    /**
027     * Create a new Coordinates object with specified WGS-84 latitude and
028     * longitude.
029     *
030     * @param latitude  latitude
031     * @param longitude longitude
032     * @return Coordinates object
033     */
034    public static Coordinates create(double latitude, double longitude) {
035        return new AutoValue_Coordinates(latitude, longitude);
036    }
037
038    /**
039     * Get a latitude
040     *
041     * @return latitude
042     */
043    public abstract double latitude();
044
045    /**
046     * Get a longitude
047     *
048     * @return longitude
049     */
050    public abstract double longitude();
051
052    /**
053     * Parse coordinates from text representation of latitude and longitude. If
054     * error occurs during parsing coordinates (e.g. wrong format) return null.
055     *
056     * @param latitude  latitude
057     * @param longitude longitude
058     * @return Coordinates object with parsed latitude and longitude
059     * @see CoordinatesParser#parse(String, String)
060     */
061    public static Coordinates parseCoordinates(String latitude, String longitude) {
062        try {
063            return CoordinatesParser.parse(latitude, longitude);
064        } catch (ParseException e) {
065            return null;
066        }
067    }
068
069    /**
070     * Parse coordinates from text representation of latitude and longitude. If
071     * error occurs during parsing coordinates (e.g. wrong format) return null.
072     * Safe flag can change behavior of parser to support coordinates without
073     * cardinal point.
074     *
075     * @param latitude  latitude
076     * @param longitude longitude
077     * @param safe      use safe parsing (mean coordinates must start with cardinal point)
078     *                  or not (allow to use negative char behind degree)
079     * @return Coordinates object with parsed latitude and longitude
080     * @see CoordinatesParser#parse(String, String, boolean)
081     */
082    public static Coordinates parseCoordinates(String latitude, String longitude, boolean safe) {
083        try {
084            return CoordinatesParser.parse(latitude, longitude, safe);
085        } catch (ParseException e) {
086            return null;
087        }
088    }
089
090    /**
091     * Parse coordinates from text representation. If error occurs during parsing
092     * coordinates (e.g. wrong format) return null.
093     *
094     * @param coordinates coordinates in a text representation
095     * @return Coordinates object with parsed latitude and longitude
096     * @see CoordinatesParser#parse(String)
097     */
098    public static Coordinates parseCoordinates(String coordinates) {
099        try {
100            return CoordinatesParser.parse(coordinates);
101        } catch (ParseException e) {
102            return null;
103        }
104    }
105
106    /**
107     * Return String representation of coordinates using specified
108     * CoordinatesFormatter instance.
109     *
110     * @param format instance of CoordinatesFormatter object used to formating
111     *               coordinates
112     * @return formatted string
113     */
114    public String toString(CoordinatesFormatter format) {
115        return format.format(this);
116    }
117
118    /**
119     * Return String representation of coordinates using specified
120     * CoordinatesFormatter format constant.
121     *
122     * @param format format constant from CoordinatesFormatter
123     * @return formatted string
124     */
125    public String toString(int format) {
126        return new CoordinatesFormatter(format).format(this);
127    }
128
129    @Override
130    public String toString() {
131        return toString(LAT_LON_DECMINUTE_FORMAT);
132    }
133
134    /**
135     * Get a distance in meters between this coordinates and specified coordinates
136     *
137     * @param to Coordinates object
138     * @return distance in meters
139     * @since 1.20
140     */
141    public double distanceTo(Coordinates to) {
142        double[] results = new double[1];
143
144        computeDistanceAndBearing(this, to, results);
145
146        return results[0];
147    }
148
149    /**
150     * <p>Computes the approximate distance in meters between two coordinates, and
151     * optionally the initial and final bearings bearings of the shortest path between
152     * them.
153     * </p>
154     * <p>
155     * The computed distance is stored in <code>results[0]</code>. If results has length 2,
156     * the initial bearing is stored in <code>results[1]</code>. If results has
157     * length 3, the final bearing is stored in <code>results[2]</code>.
158     * </p>
159     * <p>
160     * Compute is based on <a href="http://en.wikipedia.org/wiki/Haversine_formula">Haversine formula</a>.
161     * Precision is around 99.9%.
162     * </p>
163     *
164     * @param source      source coordinates
165     * @param destination destination coordinates
166     * @param results     array where first index is distance in meters, second index is initial bearing in degree
167     *                    and third index is final bearing in degree
168     * @since 1.20
169     */
170    public static void computeDistanceAndBearing(Coordinates source, Coordinates destination, double[] results) {
171        if (results == null || results.length < 1 || results.length > 3)
172            throw new IllegalArgumentException("Results has to be initialized array of size 1, 2 or 3.");
173
174        double lat1 = Math.toRadians(source.latitude());
175        double lon1 = Math.toRadians(source.longitude());
176        double lat2 = Math.toRadians(destination.latitude());
177        double lon2 = Math.toRadians(destination.longitude());
178
179        // prepare variables
180        double cosLat1 = Math.cos(lat1);
181        double cosLat2 = Math.cos(lat2);
182        double sinDLat2 = Math.sin((lat2 - lat1) / 2.0);
183        double sinDLon2 = Math.sin((lon2 - lon1) / 2.0);
184
185        // compute values
186        double a = sinDLat2 * sinDLat2 + cosLat1 * cosLat2 * sinDLon2 * sinDLon2;
187        double d = 2.0 * Math.atan2(Math.sqrt(a), Math.sqrt(1.0 - a));
188
189        // convert to metres
190        results[0] = d * AVERAGE_RADIUS_OF_EARTH;
191
192        // compute initial bearing
193        if (results.length > 1) {
194            double sinLambda = Math.sin(lon2 - lon1);
195            double cosLambda = Math.cos(lon2 - lon1);
196
197            double y = sinLambda * cosLat2;
198            double x = cosLat1 * Math.sin(lat2) - Math.sin(lat1) * cosLat2 * cosLambda;
199            results[1] = Math.toDegrees(Math.atan2(y, x));
200
201            // compute final bearing
202            if (results.length > 2) {
203                y = sinLambda * cosLat1;
204                x = -Math.sin(lat1) * cosLat2 + cosLat1 * Math.sin(lat2) * cosLambda;
205                results[2] = Math.toDegrees(Math.atan2(y, x));
206            }
207        }
208    }
209
210    @Override
211    public boolean equals(Object obj) {
212        return obj instanceof Coordinates && Math.abs(((Coordinates) obj).latitude() - latitude()) < PRECISION && Math.abs(((Coordinates) obj).longitude() - longitude()) < PRECISION;
213    }
214
215    public static Builder builder() {
216        return new Builder();
217    }
218
219    public static class Builder {
220        private double latitude = Double.NaN;
221        private double longitude = Double.NaN;
222
223        Builder() {
224        }
225
226        public Builder latitude(double latitude) {
227            this.latitude = latitude;
228            return this;
229        }
230
231        public Builder longitude(double longitude) {
232            this.longitude = longitude;
233            return this;
234        }
235
236        @Nullable
237        public Coordinates build() {
238            if (Double.isNaN(latitude) || Double.isNaN(longitude))
239                return null;
240
241            return Coordinates.create(latitude, longitude);
242        }
243    }
