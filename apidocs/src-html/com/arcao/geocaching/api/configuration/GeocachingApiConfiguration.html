001package com.arcao.geocaching.api.configuration;
002
003import com.arcao.geocaching.api.configuration.impl.DefaultProductionGeocachingApiConfiguration;
004import com.arcao.geocaching.api.configuration.impl.DefaultStagingGeocachingApiConfiguration;
005
006import org.jetbrains.annotations.NotNull;
007
008/**
009 * Basic interface for configuration class
010 *
011 * @author arcao
012 */
013public interface GeocachingApiConfiguration {
014    /**
015     * Production configuration
016     */
017    GeocachingApiConfiguration PRODUCTION = new DefaultProductionGeocachingApiConfiguration();
018    /**
019     * Staging configuration
020     */
021    GeocachingApiConfiguration STAGING = new DefaultStagingGeocachingApiConfiguration();
022
023    /**
024     * Returns a common part of Gecaching API service URL
025     *
026     * @return url
027     */
028    @NotNull
029    String getApiServiceEntryPointUrl();
030
031    /**
032     * Sets a specified timeout value, in milliseconds, to be used when opening
033     * a communications link. A timeout of zero is interpreted as an infinite timeout.
034     *
035     * @return an <code>int</code> that specifies the connect timeout value in milliseconds
036     * @since 1.5.16
037     */
038    int getConnectTimeout();
039
040    /**
041     * Sets the read timeout to a specified timeout, in milliseconds. A non-zero
042     * value specifies the timeout when reading from Input stream when a connection is
043     * established to a resource. A timeout of zero is interpreted as an infinite timeout.
044     *
045     * @return an <code>int</code> that specifies the timeout value to be used in milliseconds
046     * @since 1.5.16
047     */
048    int getReadTimeout();
049}
