001package com.arcao.geocaching.api;
002
003import com.arcao.geocaching.api.data.DeviceInfo;
004import com.arcao.geocaching.api.data.FieldNote;
005import com.arcao.geocaching.api.data.Geocache;
006import com.arcao.geocaching.api.data.GeocacheLimits;
007import com.arcao.geocaching.api.data.GeocacheLog;
008import com.arcao.geocaching.api.data.GeocacheStatus;
009import com.arcao.geocaching.api.data.ImageData;
010import com.arcao.geocaching.api.data.Trackable;
011import com.arcao.geocaching.api.data.TrackableLog;
012import com.arcao.geocaching.api.data.TrackableTravel;
013import com.arcao.geocaching.api.data.UserProfile;
014import com.arcao.geocaching.api.data.apilimits.ApiLimitsResponse;
015import com.arcao.geocaching.api.data.bookmarks.Bookmark;
016import com.arcao.geocaching.api.data.bookmarks.BookmarkList;
017import com.arcao.geocaching.api.data.sort.SortBy;
018import com.arcao.geocaching.api.data.type.GeocacheLogType;
019import com.arcao.geocaching.api.exception.GeocachingApiException;
020import com.arcao.geocaching.api.filter.Filter;
021
022import org.jetbrains.annotations.NotNull;
023import org.jetbrains.annotations.Nullable;
024
025import java.util.Date;
026import java.util.List;
027
028/**
029 * Interface of Geocaching API
030 *
031 * @author arcao
032 * @since 1.0
033 */
034public interface GeocachingApi {
035    enum ResultQuality {
036        SUMMARY,
037        LITE,
038        FULL
039    }
040
041
042    /**
043     * Gets a session id for current logged user
044     *
045     * @return session id
046     * @since 1.0
047     */
048    @Nullable
049    String getSession();
050
051    /**
052     * Open a session with giving session id of already logged-in user
053     *
054     * @param session session id
055     * @throws GeocachingApiException If Geocaching API error occurs
056     * @since 1.0
057     */
058    void openSession(@NotNull String session) throws GeocachingApiException;
059
060    /**
061     * Close current used session
062     *
063     * @since 1.0
064     */
065    void closeSession();
066
067    /**
068     * Check if current used session is still valid
069     *
070     * @return true if is valid otherwise false
071     * @since 1.0
072     */
073    boolean isSessionValid();
074
075    /**
076     * Get a trackable object for given trackable code. This method works for
077     * public trackable code and also for private (secret) trackable code.
078     *
079     * @param trackableCode     trackable code
080     * @param trackableLogCount count of trackable logs to get
081     * @return information about trackable
082     * @throws GeocachingApiException If error occurs during getting information
083     * @since 1.1
084     */
085    @Nullable
086    Trackable getTrackable(@NotNull String trackableCode, int trackableLogCount) throws GeocachingApiException;
087
088    /**
089     * Get a list of trackables which is currently placed in a cache.
090     *
091     * @param cacheCode         cache code
092     * @param startIndex        count of trackables to skip
093     * @param maxPerPage        count of trackables to get
094     * @param trackableLogCount count of trackable logs to get
095     * @return list of trackables
096     * @throws GeocachingApiException If error occurs during getting information
097     * @since 1.1
098     */
099    @NotNull
100    List<Trackable> getTrackablesByCacheCode(@NotNull String cacheCode, int startIndex, int maxPerPage, int trackableLogCount) throws GeocachingApiException;
101
102    /**
103     * Get a list of complete travel track of specified trackable.
104     *
105     * @param trackableCode trackable code
106     * @return information about trackable travel track
107     * @throws GeocachingApiException If error occurs during getting information
108     * @since 1.20
109     */
110    @NotNull
111    List<TrackableTravel> getTrackableTravelList(@NotNull String trackableCode) throws GeocachingApiException;
112
113    /**
114     * Get a list of geocache logs in given cache.
115     *
116     * @param cacheCode  cache code
117     * @param startIndex count of logs to skip
118     * @param maxPerPage count of logs to get
119     * @return list of cache logs
120     * @throws GeocachingApiException If error occurs during getting information
121     * @since 1.1
122     */
123    @NotNull
124    List<GeocacheLog> getGeocacheLogsByCacheCode(@NotNull String cacheCode, int startIndex, int maxPerPage) throws GeocachingApiException;
125
126    /**
127     * Get an information about geocache.
128     *
129     * @param resultQuality     How much data will be returned
130     * @param cacheCode         cache code
131     * @param cacheLogCount     count of logs to get
132     * @param trackableLogCount count of trackable logs to get
133     * @return full cache information
134     * @throws GeocachingApiException If error occurs during getting information
135     * @since 1.0
136     */
137    @Nullable
138    Geocache getGeocache(@NotNull ResultQuality resultQuality, @NotNull String cacheCode, int cacheLogCount, int trackableLogCount) throws GeocachingApiException;
139
140    /**
141     * Search for geocaches and return list of found.
142     *
143     * @param resultQuality     How much data will be returned
144     * @param maxPerPage        count of caches to get
145     * @param geocacheLogCount  count of logs to get
146     * @param trackableLogCount count of trackables to get
147     * @param filters           used filters while searching
148     * @param sortByList        sorting result list configuration
149     * @return list of found caches
150     * @throws GeocachingApiException If error occurs during searching caches
151     * @since 1.6
152     */
153    @NotNull
154    List<Geocache> searchForGeocaches(@NotNull ResultQuality resultQuality, int maxPerPage, int geocacheLogCount, int trackableLogCount, @NotNull List<Filter> filters, @Nullable List<SortBy> sortByList) throws GeocachingApiException;
155
156    /**
157     * Retrieve next geocaches searched by searchForGeocaches method.
158     *
159     * @param resultQuality     How much data will be returned
160     * @param startIndex        count of caches to skip
161     * @param maxPerPage        count of caches to get
162     * @param geocacheLogCount  count of caches to get
163     * @param trackableLogCount count of trackables to get
164     * @return list of found caches
165     * @throws GeocachingApiException If error occurs during searching caches
166     * @since 1.6
167     */
168    @NotNull
169    List<Geocache> getMoreGeocaches(@NotNull ResultQuality resultQuality, int startIndex, int maxPerPage, int geocacheLogCount, int trackableLogCount) throws GeocachingApiException;
170
171    /**
172     * Get an information about user
173     *
174     * @param challengesData    include challanges data
175     * @param favoritePointData include favorites points
176     * @param geocacheData      include information about caches
177     * @param publicProfileData include public profile information
178     * @param souvenirData      include souvenirs
179     * @param trackableData     include trackables
180     * @param deviceInfo        information about used device
181     * @return UserProfile object with selected information
182     * @throws GeocachingApiException If error occurs during getting information
183     * @since 1.2
184     */
185    @Nullable
186    UserProfile getYourUserProfile(boolean challengesData, boolean favoritePointData, boolean geocacheData, boolean publicProfileData, boolean souvenirData, boolean trackableData, @NotNull DeviceInfo deviceInfo) throws GeocachingApiException;
187
188    /**
189     * Create field note and publish them or store them to list of Field notes
190     * on Geocaching site.
191     *
192     * @param cacheCode         geocache which can own this field note
193     * @param geocacheLogType   type of log
194     * @param dateLogged        when cache was found
195     * @param note              text of field note
196     * @param publish           true if publish log to cache or false to add to Field note
197     *                          list on Geocaching site
198     * @param imageData         data about image to publish together or null to not publish
199     *                          any image
200     * @param favoriteThisCache true if add cache to favorite list otherwise false
201     * @return cache log
202     * @throws GeocachingApiException If error occurs during sending field note
203     * @since 1.1
204     */
205    @Nullable
206    GeocacheLog createFieldNoteAndPublish(@NotNull String cacheCode, @NotNull GeocacheLogType geocacheLogType, @NotNull Date dateLogged, @NotNull String note, boolean publish, @Nullable ImageData imageData, boolean favoriteThisCache) throws GeocachingApiException;
207
208    /**
209     * Create field note and publish them or store them to list of Field notes
210     * on Geocaching site.
211     *
212     * @param fieldNote         Field note object
213     * @param publish           true if publish log to cache or false to add to Field note
214     *                          list on Geocaching site
215     * @param imageData         data about image to publish together or null to not publish
216     *                          any image
217     * @param favoriteThisCache true if add cache to favorite list otherwise false
218     * @return cache log
219     * @throws GeocachingApiException If error occurs during sending field note
220     * @since 1.1
221     */
222    @Nullable
223    GeocacheLog createFieldNoteAndPublish(@NotNull FieldNote fieldNote, boolean publish, @Nullable ImageData imageData, boolean favoriteThisCache) throws GeocachingApiException;
224
225    /**
226     * Store a personal note for specified geocache. If note parameter is null or
227     * empty the note will be removed.
228     *
229     * @param cacheCode geocache which can own this personal note
230     * @param note      personal note
231     * @throws GeocachingApiException If error occurs during personal note storing
232     * @since 1.4.2
233     */
234    void setGeocachePersonalNote(@NotNull String cacheCode, @Nullable String note) throws GeocachingApiException;
235
236    /**
237     * Remove existing personal note from a specified geocache
238     *
239     * @param cacheCode geocache where the personal note has to be deleted
240     * @throws GeocachingApiException If error occurs during personal note storing
241     * @since 1.6.4
242     */
243    void deleteCachePersonalNote(@NotNull String cacheCode) throws GeocachingApiException;
244
245    /**
246     * Get a list of trackable logs for given public trackable code.
247     *
248     * @param trackableCode public trackable code
249     * @param startIndex    count of trackable logs to skip
250     * @param maxPerPage    count of trackable logs to get
251     * @return list of trackable logs
252     * @throws GeocachingApiException If error occurs during getting information
253     * @since 1.5.1
254     */
255    @NotNull
256    List<TrackableLog> getTrackableLogs(@NotNull String trackableCode, int startIndex, int maxPerPage) throws GeocachingApiException;
257
258    /**
259     * Returns the API limits applied on currently logged user or null if this
260     * information isn't available.
261     *
262     * @return API limits response object
263     * @throws GeocachingApiException If error occurs during getting information
264     * @since 1.5.10
265     */
266    @NotNull
267    ApiLimitsResponse getApiLimits() throws GeocachingApiException;
268
269    /**
270     * Return the cache limits received with a last Geocache request. If the
271     * last request was not Geocache request, returns null.
272     *
273     * @return cache limits or null
274     * @since 1.5.10
275     */
276    @Nullable
277    GeocacheLimits getLastGeocacheLimits();
278
279    /**
280     * Return count of Geocaches found in a last Geocache request. If the
281     * last request was not Geocache request, returns <tt>0</tt>.
282     *
283     * @return count of Geocaches found
284     * @since 1.6
285     */
286    int getLastSearchResultsFound();
287
288    /**
289     * Get a list of users geocache logs
290     *
291     * @param userName        user name
292     * @param startDate       start date of created geocache log or null
293     * @param endDate         end date of created geocache log or null
294     * @param logTypes        array of geocache log types to filter (must contains at least one Geocache log type)
295     * @param excludeArchived exclude archived geocaches
296     * @param startIndex      count of logs to skip
297     * @param maxPerPage      count of logs to get
298     * @return list of cache logs
299     * @throws GeocachingApiException If error occurs during getting information
300     * @since 1.5.13
301     */
302    @NotNull
303    List<GeocacheLog> getUsersGeocacheLogs(@NotNull String userName, @Nullable Date startDate, @Nullable Date endDate, @NotNull GeocacheLogType[] logTypes, boolean excludeArchived, int startIndex, int maxPerPage) throws GeocachingApiException;
304
305    /**
306     * Get a user's owned trackables
307     *
308     * @param startIndex        count of trackables to skip
309     * @param maxPerPage        count of trackables to get
310     * @param trackableLogCount count of trackables to get
311     * @param collectionOnly    gets only trackables which are in collection
312     * @return list of trackables
313     * @throws GeocachingApiException If error occurs during getting information
314     */
315    @NotNull
316    List<Trackable> getUsersTrackables(int startIndex, int maxPerPage, int trackableLogCount, boolean collectionOnly) throws GeocachingApiException;
317
318    /**
319     * Returns a list of Bookmark lists for current user.
320     *
321     * @return list of Bookmark lists
322     * @throws GeocachingApiException If error occurs during getting information
323     * @since 1.6.2
324     */
325    @NotNull
326    List<BookmarkList> getBookmarkListsForUser() throws GeocachingApiException;
327
328    /**
329     * Return a list of Bookmark lists for specified user id
330     *
331     * @param userId numeric user id from User object
332     * @return list of Bookmark lists
333     * @throws GeocachingApiException If error occurs during getting information
334     * @since 1.6.2
335     */
336    @NotNull
337    List<BookmarkList> getBookmarkListsByUserId(int userId) throws GeocachingApiException;
338
339    /**
340     * Retrieve a list of Bookmarked caches by a guid
341     *
342     * @param guid guid of that list
343     * @return list of Bookmarked caches
344     * @throws GeocachingApiException If error occurs during getting information
345     * @since 1.6.2
346     */
347    @NotNull
348    List<Bookmark> getBookmarkListByGuid(String guid) throws GeocachingApiException;
349
350    /**
351     * Retrieve a list of Geocache Status
352     *
353     * @param cacheCodes list of Geocache codes
354     * @return list of Geocache status
355     * @throws GeocachingApiException If error occurs during getting information
356     * @since 1.6.3
357     */
358    @NotNull
359    List<GeocacheStatus> getGeocacheStatus(@NotNull List<String> cacheCodes) throws GeocachingApiException;
360
361}
