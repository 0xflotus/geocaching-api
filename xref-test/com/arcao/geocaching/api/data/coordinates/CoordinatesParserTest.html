3   import org.junit.Test;

5   import java.text.ParseException;

7   import static org.junit.Assert.assertEquals;

11  public class CoordinatesParserTest {
12    private static final double refLongitude = 8.0 + 38.564 / 60.0;
13    private static final double refLatitude = 49.0 + 56.031 / 60.0;
14    private static final Coordinates refCoordinates = Coordinates.create(refLatitude, refLongitude);
15    private static final double PRECISION = 1e-8;
16  
17    @Test
18    public void testParseLatitude() throws ParseException {
19      assertEquals(refLatitude, CoordinatesParser.parseLatitude("N 49° 56.031"), PRECISION);
20    }
21  
22    @Test
23    public void testParseLongitude() throws ParseException {
24      assertEquals(refLongitude, CoordinatesParser.parseLongitude("E 8° 38.564"), PRECISION);
25    }
26  
27    @Test
28    public void testFullCoordinates() throws ParseException {
29      assertTrue(refCoordinates.equals(Coordinates.parseCoordinates("N 49° 56.031 | E 8° 38.564")));
30    }
31  
32    @Test
33    public void testCoordinateMissingPart() {
34      // we are trying to parse a _point_, but have only a latitude. Don't accept
35      // the numerical part as longitude!
36      Coordinates point = null;
37      try {
38        point = CoordinatesParser.parse("N 49° 56.031");
39      } catch (ParseException e) {
40        // expected
41      }
42      assertEquals(null, point);
43    }
44  
45    @Test
46    public void testSouth() throws ParseException {
47      assertEquals(-refLatitude, CoordinatesParser.parseLatitude("S 49° 56.031"), PRECISION);
48    }
49  
50    @Test
51    public void testWest() throws ParseException {
52      assertEquals(-refLongitude, CoordinatesParser.parseLongitude("W 8° 38.564"), PRECISION);
53    }
54  
55    @Test
56    public void testLowerCase() throws ParseException {
57      assertEquals(refLongitude, CoordinatesParser.parseLongitude("e 8° 38.564"), PRECISION);
58    }
59  
60    @Test
61    public void testVariousFormats() throws ParseException {
62      final Coordinates point1 = CoordinatesParser.parse("N 49° 43' 57\" | E 2 12' 35");
63      final Coordinates point2 = CoordinatesParser.parse("N 49 43.95 E2°12.5833333333");
64      assertTrue(point1.equals(point2));
65    }
66  
67    @Test
68    public void testInSentence() throws ParseException {
69      final Coordinates p1 = CoordinatesParser.parse("Station3: N51 21.523 / E07 02.680");
70      final Coordinates p2 = CoordinatesParser.parse("N51 21.523", "E07 02.680");
71      assertNotNull(p1);
72      assertNotNull(p2);
73      assertTrue(p1.equals(p2));
74    }
75  
76    @Test
77    public void testUnrelatedParts() {
78      Coordinates point = null;
79      try {
80        point = CoordinatesParser.parse("N51 21.523 and some words in between, so there is no relation E07 02.680");
81      } catch (ParseException e) {
82        // expected
83      }
84      assertEquals(null, point);
85    }
86  
87    @Test
88    public void testComma() throws ParseException {
89      final Coordinates pointComma = CoordinatesParser.parse("N 46° 27' 55,65''\n" +
90          "E 15° 53' 41,68''");
91      final Coordinates pointDot = CoordinatesParser.parse("N 46° 27' 55.65''\n" +
92          "E 15° 53' 41.68''");
93      assertNotNull(pointComma);
94      assertNotNull(pointDot);
95      assertTrue(pointComma.equals(pointDot));
96    }
97    
98    @Test
99    public void testWithoutCardinalPoint() throws ParseException {
100     assertEquals(refLatitude, CoordinatesParser.parseLatitude("49° 56.031", false), PRECISION);
101     assertEquals(refLongitude, CoordinatesParser.parseLongitude("8° 38.564", false), PRECISION);
102     
103     assertEquals(-refLatitude, CoordinatesParser.parseLatitude("-49° 56.031", false), PRECISION);
104     assertEquals(-refLongitude, CoordinatesParser.parseLongitude("-8° 38.564", false), PRECISION);
105   }
