11  public class CoordinatesFormatterTest {
12    private static final double refLatitude = 49.0 + 56.031 / 60.0;
13    private static final double refLongitude = 8.0 + 38.564 / 60.0;
14    private static final Coordinates refCoordinates = Coordinates.create(refLatitude, refLongitude);
15    private static final char decSymbol = new DecimalFormatSymbols().getDecimalSeparator(); // some tests are Locale specific!!!
16    
17    @Test
18    public void testLatLonDecDegree() throws ParseException {
19      CoordinatesFormatter f = new CoordinatesFormatter(CoordinatesFormatter.LAT_LON_DECDEGREE);
20      assertEquals("49" + decSymbol + "933850 8" + decSymbol + "642733", f.format(refCoordinates));
21    }
22    
23    @Test
24    public void testLatLonDecDegreeComma() throws ParseException {

27    }
