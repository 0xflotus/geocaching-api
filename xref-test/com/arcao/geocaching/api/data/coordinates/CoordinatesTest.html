11  public class CoordinatesTest {
12    private static final double refLongitude = 8.0 + 38.564 / 60.0;
13    private static final double refLatitude = 49.0 + 56.031 / 60.0;
14    private static final Coordinates refCoordinates = Coordinates.create(refLatitude, refLongitude);
15  
16    @Test
17    public void testCoordinates() throws ParseException {
18      assertTrue(refCoordinates.equals(Coordinates.parseCoordinates("N 49° 56.031 | E 8° 38.564")));
19    }
20  
21    @Test
22    public void testCoordinateMissingPart() {
23      // we are trying to parse a _point_, but have only a latitude. Don't accept
24      // the numerical part as longitude!
25      Coordinates point = Coordinates.parseCoordinates("N 49° 56.031");
26      assertNull(point);
27    }
28  
29    @Test
30    public void testVariousFormats() throws ParseException {
31      final Coordinates point1 = Coordinates.parseCoordinates("N 49° 43' 57\" | E 2 12' 35");
32      final Coordinates point2 = Coordinates.parseCoordinates("N 49 43.95 E2°12.5833333333");
33  
34      assertNotNull(point1);
35      assertTrue(point1.equals(point2));
36    }
37  
38    @Test
39    public void testLatitudeLongitude() throws ParseException {
40      final Coordinates point2 = Coordinates.parseCoordinates("N 49° 56.031", "E 8° 38.564");
41  
42      assertTrue(refCoordinates.equals(point2));
43    }
44  }
