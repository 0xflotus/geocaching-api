25          throw e;

29    @Test
30    public void simpleSetGeocachePersonalNoteTest() throws GeocachingApiException, InterruptedException {
31      final String expected = "Test1234";
32  
33      // race condition on api.geocaching.com server: wait for deletion
34      Thread.sleep(5000);
35  
36      Geocache geocache =  api.getGeocache(GeocachingApi.ResultQuality.FULL, CACHE_CODE, 0, 0);
37      Assert.assertNotNull(geocache);
38      Assert.assertThat(geocache.personalNote(), isEmptyOrNullString());
39  
40      api.setGeocachePersonalNote(CACHE_CODE, expected);
41  
42      // race condition on api.geocaching.com server: wait for setting value
43      Thread.sleep(5000);
44  
45      geocache = api.getGeocache(GeocachingApi.ResultQuality.FULL, CACHE_CODE, 0, 0);
46      Assert.assertNotNull(geocache);
47      Assert.assertEquals(expected, geocache.personalNote());
